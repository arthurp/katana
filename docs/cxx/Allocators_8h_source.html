175 public:

213  struct FreeNode {
214  FreeNode* next;

225  } while (!__sync_bool_compare_and_swap(&head, h, 0));
226  while (h) {

236  inline void* allocate(size_t size) {
237  static SimpleLock lock;
238 
239  lock.lock();
240  FreeNode* OH = 0;
241  FreeNode* NH = 0;
242  do {
243  OH = head;
244  if (!OH) {
245  lock.unlock();
246  return SourceHeap::allocate(size);
247  }
248  NH = OH->next; // The lock protects this line
249  } while (!__sync_bool_compare_and_swap(&head, OH, NH));
250  lock.unlock();
251  KATANA_LOG_DEBUG_ASSERT(OH);
252  return (void*)OH;
253  }

255  inline void deallocate(void* ptr) {
256  if (!ptr)
257  return;
258  FreeNode* OH;
259  FreeNode* NH;
260  do {
261  OH = head;
262  NH = (FreeNode*)ptr;
263  NH->next = OH;
264  } while (!__sync_bool_compare_and_swap(&head, OH, NH));
265  }

268 template <unsigned ElemSize, typename SourceHeap>
269 class BlockHeap : public SourceHeap {
270  struct TyEq {
271  double data
272  [((ElemSize + sizeof(double) - 1) & ~(sizeof(double) - 1)) /
273  sizeof(double)];
274  };
275 
276  struct Block_basic {
277  union {
278  Block_basic* next;
279  double dummy;
280  };
281  TyEq data[1];
282  };
283 
284  enum {
285  BytesLeft = (SourceHeap::AllocSize - sizeof(Block_basic)),
286  BytesLeftR = BytesLeft & ~(sizeof(double) - 1),
287  FitLeft = BytesLeftR / sizeof(TyEq[1]),
288  TotalFit = FitLeft + 1
289  };
290 
291  struct Block {
292  union {
293  Block* next;
294  double dummy;
295  };
296  TyEq data[TotalFit];
297  };
298 
299  Block* head;
300  int headIndex;
301 
302  void refill() {
303  void* P = SourceHeap::allocate(SourceHeap::AllocSize);
304  Block* BP = (Block*)P;
305  BP->next = head;
306  head = BP;
307  headIndex = 0;
308  }
309 
310 public:
311  enum { AllocSize = ElemSize };
312 
313  void clear() {
314  while (head) {
315  Block* B = head;
316  head = B->next;
317  SourceHeap::deallocate(B);
318  }
319  }
320 
321  BlockHeap() : SourceHeap(), head(0), headIndex(0) {
322  static_assert(sizeof(Block) <= SourceHeap::AllocSize, "");
323  }
324 
325  ~BlockHeap() { clear(); }
326 
327  inline void* allocate([[maybe_unused]] size_t size) {
328  KATANA_LOG_DEBUG_ASSERT(size == ElemSize);
329  if (!head || headIndex == TotalFit)
330  refill();
331  return &head->data[headIndex++];
332  }
333 
334  inline void deallocate(void*) {}
335 };

338 template <typename SourceHeap>
339 class BumpHeap : public SourceHeap {
340  struct Block {
341  union {
342  Block* next;
343  double dummy; // for alignment
344  };
345  };
346 
347  Block* head;
348  int offset;
349 
350  void refill() {
351  void* P = SourceHeap::allocate(SourceHeap::AllocSize);
352  Block* BP = (Block*)P;
353  BP->next = head;
354  head = BP;
355  offset = sizeof(Block);
356  }
357 
358 public:
359  enum { AllocSize = 0 };
360 
361  BumpHeap() : SourceHeap(), head(0), offset(0) {}
362 
363  ~BumpHeap() { clear(); }
364 
365  void clear() {
366  while (head) {
367  Block* B = head;
368  head = B->next;
369  SourceHeap::deallocate(B);
370  }
371  }
372 
373  inline void* allocate(size_t size) {
374  // Increase to alignment
375  size_t alignedSize = (size + sizeof(double) - 1) & ~(sizeof(double) - 1);
376  // Check current block
377  if (!head || offset + alignedSize > SourceHeap::AllocSize) {
378  refill();
379  }
380  if (offset + alignedSize > SourceHeap::AllocSize) {
381  std::abort(); // TODO: remove
382  throw std::bad_alloc();
383  }
384  char* retval = (char*)head;
385  retval += offset;
386  offset += alignedSize;
387  return retval;
388  }
389 
394  inline void* allocate(size_t size, size_t& allocated) {
395  // Increase to alignment
396  size_t alignedSize = (size + sizeof(double) - 1) & ~(sizeof(double) - 1);
397  if (alignedSize > SourceHeap::AllocSize) {
398  alignedSize = SourceHeap::AllocSize;
399  }
400  // Check current block
401  if (!head || offset + alignedSize > SourceHeap::AllocSize) {
402  size_t remaining = SourceHeap::AllocSize - offset;
403  KATANA_LOG_DEBUG_ASSERT(
404  (remaining & (sizeof(double) - 1)) == 0); // should still be aligned
405  if (!remaining) {
406  refill();
407  } else {
408  alignedSize = remaining;
409  }
410  }
411  char* retval = (char*)head;
412  retval += offset;
413  offset += alignedSize;
414  allocated = (alignedSize > size) ? size : alignedSize;
415  return retval;
416  }
417 
418  inline void deallocate(void*) {}
419 };

425 template <typename SourceHeap>
426 class BumpWithMallocHeap : public SourceHeap {
427  struct Block {
428  union {
429  Block* next;
430  double dummy; // for alignment
431  };
432  };
433 
434  Block* head;
435  Block* fallbackHead;
436  int offset;
437 
439  void refill(void* P, Block*& h, int* o) {
440  Block* BP = (Block*)P;
441  BP->next = h;
442  h = BP;
443  if (o)
444  *o = sizeof(Block);
445  }
446 
447 public:
448  enum { AllocSize = 0 };
449 
450  BumpWithMallocHeap() : SourceHeap(), head(0), fallbackHead(0), offset(0) {}
451 
452  ~BumpWithMallocHeap() { clear(); }
453 
454  void clear() {
455  while (head) {
456  Block* B = head;
457  head = B->next;
458  SourceHeap::deallocate(B);
459  }
460  while (fallbackHead) {
461  Block* B = fallbackHead;
462  fallbackHead = B->next;
463  free(B);
464  }
465  }
466 
467  inline void* allocate(size_t size) {
468  // Increase to alignment
469  size_t alignedSize = (size + sizeof(double) - 1) & ~(sizeof(double) - 1);
470  if (sizeof(Block) + alignedSize > SourceHeap::AllocSize) {
471  void* p = malloc(alignedSize + sizeof(Block));
472  refill(p, fallbackHead, NULL);
473  return (char*)p + sizeof(Block);
474  }
475  // Check current block
476  if (!head || offset + alignedSize > SourceHeap::AllocSize)
477  refill(SourceHeap::allocate(SourceHeap::AllocSize), head, &offset);
478  char* retval = (char*)head;
479  retval += offset;
480  offset += alignedSize;
481  return retval;
482  }
483 
484  inline void deallocate(void*) {}
485 };

744 class KATANA_EXPORT Pow2BlockHeap : public StaticSingleInstance<Pow2BlockHeap> {
745 private:
746  using Base = StaticSingleInstance<Pow2BlockHeap>;
747  /* template <typename> */ friend class StaticSingleInstance<Pow2BlockHeap>;
748 
749  static const bool USE_MALLOC_AS_BACKUP = true;
750 
751  static const size_t LOG2_MIN_SIZE = 3; // 2^3 == 8 bytes
752  static const size_t LOG2_MAX_SIZE = 16; // 64k
753 
754  typedef FixedSizeHeap Heap_ty;
755 
756  std::vector<Heap_ty> heapTable;
757 
758  static inline size_t pow2(unsigned i) { return (1U << i); }
759 
760  static unsigned nextLog2(const size_t allocSize) {
761  unsigned i = LOG2_MIN_SIZE;
762 
763  while (pow2(i) < allocSize) {
764  ++i;
765  }
766 
767  // if (pow2 (i) > pow2 (LOG2_MAX_SIZE)) {
768  // std::fprintf (stderr, "ERROR: block bigger than huge page size
769  // requested\n"); throw std::bad_alloc();
770  // }
771 
772  return i;
773  }
774 
775  void populateTable(void) {
776  KATANA_LOG_DEBUG_ASSERT(heapTable.empty());
777 
778  heapTable.clear();
779  for (unsigned i = 0; i <= LOG2_MAX_SIZE; ++i) {
780  heapTable.push_back(Heap_ty(pow2(i)));
781  }
782  }
783 
784  Pow2BlockHeap() noexcept; // NOLINT(modernize-use-equals-delete)
785 
786 public:
787  void* allocateBlock(const size_t allocSize) {
788  if (allocSize > pow2(LOG2_MAX_SIZE)) {
789  if (USE_MALLOC_AS_BACKUP) {
790  return malloc(allocSize);
791  } else {
792  fprintf(stderr, "ERROR: block bigger than huge page size requested\n");
793  throw std::bad_alloc();
794  }
795  } else {
796  unsigned i = nextLog2(allocSize);
797  KATANA_LOG_DEBUG_ASSERT(i < heapTable.size());
798  return heapTable[i].allocate(pow2(i));
799  }
800  }
801 
802  void deallocateBlock(void* ptr, const size_t allocSize) {
803  if (allocSize > pow2(LOG2_MAX_SIZE)) {
804  if (USE_MALLOC_AS_BACKUP) {
805  free(ptr);
806  } else {
807  fprintf(stderr, "ERROR: block bigger than huge page size requested\n");
808  throw std::bad_alloc();
809  }
810  } else {
811  unsigned i = nextLog2(allocSize);
812  KATANA_LOG_DEBUG_ASSERT(i < heapTable.size());
813  heapTable[i].deallocate(ptr);
814  }
815  }
816 };
