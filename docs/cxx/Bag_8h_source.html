41 template <typename T, unsigned int BlockSize = 0>
42 class InsertBag {
43  struct header {
44  header* next;
45  T* dbegin; // start of interesting data
46  T* dend; // end of valid data
47  T* dlast; // end of storage
48  };

50  typedef std::pair<header*, header*> PerThread;

52 public:
53  template <typename U>
54  class Iterator : public boost::iterator_facade<
55  Iterator<U>, U, boost::forward_traversal_tag> {
56  friend class boost::iterator_core_access;
57  using Storage = katana::PerThreadStorage<std::pair<header*, header*>>;
58  using Head = std::conditional_t<std::is_const_v<U>, const Storage, Storage>;

60  Head* hd;
61  unsigned int thr;

63  U* v;

65  bool init_thread() {
66  p = thr < hd->size() ? hd->getRemote(thr)->first : 0;
67  v = p ? p->dbegin : 0;
68  return p;
69  }
70 
71  bool advance_local() {
72  if (p) {
73  ++v;
74  return v != p->dend;
75  }
76  return false;
77  }
78 
79  bool advance_chunk() {
80  if (p) {
81  p = p->next;
82  v = p ? p->dbegin : 0;
83  }
84  return p;
85  }
86 
87  void advance_thread() {
88  while (thr < hd->size()) {
89  ++thr;
90  if (init_thread())
91  return;
92  }
93  }
94 
95  void increment() {
96  if (advance_local())
97  return;
98  if (advance_chunk())
99  return;
100  advance_thread();
101  }

128  void insHeader(header* h) {
129  PerThread& hpair = *heads.getLocal();
130  if (hpair.second) {
131  hpair.second->next = h;
132  hpair.second = h;
133  } else {
134  hpair.first = hpair.second = h;
135  }
136  }

138  header* newHeaderFromHeap(void* m, unsigned size) {
139  header* H = new (m) header();
140  int offset = 1;
141  if (sizeof(T) < sizeof(header))
142  offset += sizeof(header) / sizeof(T);
143  T* a = reinterpret_cast<T*>(m);
144  H->dbegin = &a[offset];
145  H->dend = H->dbegin;
146  H->dlast = &a[(size / sizeof(T))];
147  H->next = 0;
148  return H;
149  }

151  header* newHeader() {
152  if (BlockSize) {
153  return newHeaderFromHeap(heap.allocate(BlockSize), BlockSize);
154  } else {
155  return newHeaderFromHeap(katana::pagePoolAlloc(), katana::allocSize());
156  }
157  }

159  void destruct_serial() {
160  for (unsigned x = 0; x < heads.size(); ++x) {
161  PerThread& hpair = *heads.getRemote(x);
162  header*& h = hpair.first;
163  while (h) {
164  uninitialized_destroy(h->dbegin, h->dend);
165  header* h2 = h;
166  h = h->next;
167  if (BlockSize)
168  heap.deallocate(h2);
169  else
170  katana::pagePoolFree(h2);
171  }
172  hpair.second = 0;
173  }
174  }

176  void destruct_parallel(void) {
177  katana::on_each_gen(
178  [this](const unsigned int tid, const unsigned int) {
179  PerThread& hpair = *heads.getLocal(tid);
180  header*& h = hpair.first;
181  while (h) {
182  uninitialized_destroy(h->dbegin, h->dend);
183  header* h2 = h;
184  h = h->next;
185  if (BlockSize)
186  heap.deallocate(h2);
187  else
188  katana::pagePoolFree(h2);
189  }
190  hpair.second = 0;
191  },
192  std::make_tuple(katana::no_stats()));
193  }

256  template <typename... Args>
257  reference emplace(Args&&... args) {
258  header* H = heads.getLocal()->second;
259  T* rv;
260  if (!H || H->dend == H->dlast) {
261  H = newHeader();
262  insHeader(H);
263  }
264  rv = new (H->dend) T(std::forward<Args>(args)...);
265  ++H->dend;
266  return *rv;
267  }

278  void pop() {
279  header* H = heads.getLocal()->second;
280  if (H->dbegin == H->dend) {
281  throw std::out_of_range("InsertBag::pop");
282  }
283  uninitialized_destroy(H->dend - 1, H->dend);
284  --H->dend;
285  }
