95  template <typename WL, typename TileMaker>
96  void PushEdgeTiles(WL& wl, EI beg, const EI end, const TileMaker& f) {
97  KATANA_LOG_DEBUG_ASSERT(beg <= end);
98 
99  if ((end - beg) > edge_tile_size) {
100  for (; beg + edge_tile_size < end;) {
101  auto ne = beg + edge_tile_size;
102  KATANA_LOG_DEBUG_ASSERT(ne < end);
103  wl.push(f(beg, ne));
104  beg = ne;
105  }
106  }
107 
108  if ((end - beg) > 0) {
109  wl.push(f(beg, end));
110  }
111  }

115  auto beg = graph->edge_begin(src);

117 
118  PushEdgeTiles(wl, beg, end, f);

121  template <typename WL, typename TileMaker>
122  void PushEdgeTilesParallel(
123  WL& wl, Graph* graph, GNode src, const TileMaker& f) {
124  auto beg = graph->edge_begin(src);
125  const auto end = graph->edge_end(src);
126 
127  if ((end - beg) > edge_tile_size) {
128  katana::on_each(
129  [&](const unsigned tid, const unsigned numT) {
130  auto p = katana::block_range(beg, end, tid, numT);
131 
132  auto b = p.first;
133  const auto e = p.second;
134 
135  PushEdgeTiles(wl, b, e, f);
136  },
137  katana::loopname("Init-Tiling"));
138 
139  } else if ((end - beg) > 0) {
140  wl.push(f(beg, end));
141  }

244  template <typename NodeProp, typename EdgeProp>
245  static bool Verify(Graph* graph, GNode source) {
246  if (graph->template GetData<NodeProp>(source) != 0) {
247  std::cerr << "ERROR: source has non-zero dist value == "
248  << graph->template GetData<NodeProp>(source) << std::endl;
249  return false;
250  }
251 
252  std::atomic<size_t> not_visited(0);
253  katana::do_all(katana::iterate(*graph), [&not_visited, &graph](GNode node) {
254  if (graph->template GetData<NodeProp>(node) >= kDistanceInfinity)
255  ++not_visited;
256  });
257 
258  if (not_visited)
259  std::cerr << not_visited
260  << " unvisited nodes; this is an error if the graph is "
261  "strongly connected\n";
262 
263  std::atomic<bool> not_c(false);
264  katana::do_all(
265  katana::iterate(*graph),
266  NotConsistent<NodeProp, EdgeProp>(graph, not_c));
267 
268  if (not_c) {
269  std::cerr << "node found with incorrect distance\n";
270  return false;
271  }
272 
273  katana::GReduceMax<Dist> m;
274  katana::do_all(katana::iterate(*graph), MaxDist<NodeProp>(graph, m));
275 
276  std::cout << "max dist: " << m.reduce() << "\n";
277 
278  return true;
279  }

282 template <typename T, typename BucketFunc, size_t MAX_BUCKETS = 543210ul>
283 class SerialBucketWL {
284  using Bucket = std::deque<T>;
285  using BucketsCont = std::vector<Bucket>;
286 
287  size_t m_minBucket;
288  BucketFunc m_func;
289  BucketsCont m_buckets;
290  Bucket m_lastBucket;
291 
292  static_assert(MAX_BUCKETS > 0, "MAX_BUCKETS must be > 0");
293 
294 public:
295  explicit SerialBucketWL(const BucketFunc& f) : m_minBucket(0ul), m_func(f) {
296  // reserve enough so that resize never reallocates memory
297  // otherwise, minBucket may return an invalid reference
298  m_buckets.reserve(MAX_BUCKETS);
299  }
300 
301  void push(const T& item) {
302  size_t b = m_func(item);
303  KATANA_LOG_DEBUG_VASSERT(
304  b >= m_minBucket, "can't push {} below m_minBucket {}", b, m_minBucket);
305 
306  if (b < m_buckets.size()) {
307  m_buckets[b].push_back(item);
308  return;
309  } else {
310  if (b >= MAX_BUCKETS) {
311  std::cerr << "Increase MAX_BUCKETS limit" << std::endl;
312  m_lastBucket.push_back(item);
313  } else {
314  m_buckets.resize(b + 1);
315  m_buckets[b].push_back(item);
316  }
317  }
318  }
319 
320  void goToNextBucket(void) {
321  while (m_minBucket < m_buckets.size() && m_buckets[m_minBucket].empty()) {
322  ++m_minBucket;
323  }
324  }
325 
326  Bucket& minBucket(void) {
327  if (m_minBucket < m_buckets.size()) {
328  return m_buckets[m_minBucket];
329  } else {
330  return m_lastBucket;
331  }
332  }
333 
334  bool empty(void) const { return emptyImpl(m_minBucket); }
335 
336  bool allEmpty(void) const { return emptyImpl(0ul); }
337 
338 private:
339  bool emptyImpl(size_t start) const {
340  for (size_t i = start; i < m_buckets.size(); ++i) {
341  if (!m_buckets[i].empty()) {
342  return false;
343  }
344  }
345 
346  return m_lastBucket.empty();
347  }
348 };
