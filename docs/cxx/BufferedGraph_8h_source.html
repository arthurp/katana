93  void loadOutIndex(
94  std::ifstream& graphFile, uint64_t nodeStart, uint64_t numNodesToLoad) {
95  if (numNodesToLoad == 0) {
96  return;
97  }
98  KATANA_LOG_DEBUG_ASSERT(outIndexBuffer == nullptr);
99  outIndexBuffer = (uint64_t*)malloc(sizeof(uint64_t) * numNodesToLoad);
100 
101  if (outIndexBuffer == nullptr) {
102  KATANA_DIE("Failed to allocate memory for out index buffer.");
103  }
104 
105  // position to start of contiguous chunk of nodes to read
106  uint64_t readPosition = (4 + nodeStart) * sizeof(uint64_t);
107  graphFile.seekg(readPosition);
108 
109  uint64_t numBytesToLoad = numNodesToLoad * sizeof(uint64_t);
110  uint64_t bytesRead = 0;
111 
112  while (numBytesToLoad > 0) {
113  graphFile.read(((char*)this->outIndexBuffer) + bytesRead, numBytesToLoad);
114  size_t numRead = graphFile.gcount();
115  numBytesToLoad -= numRead;
116  bytesRead += numRead;
117  }
118 
119  KATANA_LOG_DEBUG_ASSERT(numBytesToLoad == 0);
120 
121  nodeOffset = nodeStart;
122  }

133  void loadEdgeDest(
134  std::ifstream& graphFile, uint64_t edgeStart, uint64_t numEdgesToLoad,
135  uint64_t numGlobalNodes) {
136  if (numEdgesToLoad == 0) {
137  return;
138  }
139 
140  KATANA_LOG_DEBUG_ASSERT(edgeDestBuffer == nullptr);
141  edgeDestBuffer = (uint32_t*)malloc(sizeof(uint32_t) * numEdgesToLoad);
142 
143  if (edgeDestBuffer == nullptr) {
144  KATANA_DIE("Failed to allocate memory for edge dest buffer.");
145  }
146 
147  // position to start of contiguous chunk of edges to read
148  uint64_t readPosition = (4 + numGlobalNodes) * sizeof(uint64_t) +
149  (sizeof(uint32_t) * edgeStart);
150  graphFile.seekg(readPosition);
151 
152  uint64_t numBytesToLoad = numEdgesToLoad * sizeof(uint32_t);
153  uint64_t bytesRead = 0;
154  while (numBytesToLoad > 0) {
155  graphFile.read(((char*)this->edgeDestBuffer) + bytesRead, numBytesToLoad);
156  size_t numRead = graphFile.gcount();
157  numBytesToLoad -= numRead;
158  bytesRead += numRead;
159  }
160 
161  KATANA_LOG_DEBUG_ASSERT(numBytesToLoad == 0);
162  // save edge offset of this graph for later use
163  edgeOffset = edgeStart;
164  }

178  template <
179  typename EdgeType,
180  typename std::enable_if<!std::is_void<EdgeType>::value>::type* = nullptr>
181  void loadEdgeData(
182  std::ifstream& graphFile, uint64_t edgeStart, uint64_t numEdgesToLoad,
183  uint64_t numGlobalNodes, uint64_t numGlobalEdges) {
184  katana::gDebug("Loading edge data");
185 
186  if (numEdgesToLoad == 0) {
187  return;
188  }
189 
190  KATANA_LOG_DEBUG_ASSERT(edgeDataBuffer == nullptr);
191  edgeDataBuffer =
192  (EdgeDataType*)malloc(sizeof(EdgeDataType) * numEdgesToLoad);
193 
194  if (edgeDataBuffer == nullptr) {
195  KATANA_DIE("Failed to allocate memory for edge data buffer.");
196  }
197 
198  // position after nodes + edges
199  uint64_t baseReadPosition = (4 + numGlobalNodes) * sizeof(uint64_t) +
200  (sizeof(uint32_t) * numGlobalEdges);
201 
202  // version 1 padding TODO make version agnostic
203  if (numGlobalEdges % 2) {
204  baseReadPosition += sizeof(uint32_t);
205  }
206 
207  // jump to first byte of edge data
208  uint64_t readPosition =
209  baseReadPosition + (sizeof(EdgeDataType) * edgeStart);
210  graphFile.seekg(readPosition);
211  uint64_t numBytesToLoad = numEdgesToLoad * sizeof(EdgeDataType);
212  uint64_t bytesRead = 0;
213 
214  while (numBytesToLoad > 0) {
215  graphFile.read(((char*)this->edgeDataBuffer) + bytesRead, numBytesToLoad);
216  size_t numRead = graphFile.gcount();
217  numBytesToLoad -= numRead;
218  bytesRead += numRead;
219  }
220 
221  KATANA_LOG_DEBUG_ASSERT(numBytesToLoad == 0);
222  }

315  void loadGraph(const std::string& filename) {
316  if (graphLoaded) {
317  KATANA_DIE("Cannot load an buffered graph more than once.");
318  }
319 
320  std::ifstream graphFile(filename.c_str());
321  uint64_t header[4];
322  graphFile.read(((char*)header), sizeof(uint64_t) * 4);
323 
324  numLocalNodes = globalSize = header[2];
325  numLocalEdges = globalEdgeSize = header[3];
326 
327  loadOutIndex(graphFile, 0, globalSize);
328  loadEdgeDest(graphFile, 0, globalEdgeSize, globalSize);
329  // may or may not do something depending on EdgeDataType
330  loadEdgeData<EdgeDataType>(
331  graphFile, 0, globalEdgeSize, globalSize, globalEdgeSize);
332  graphLoaded = true;
333 
334  graphFile.close();
335  }

388  EdgeIterator edgeBegin(uint64_t globalNodeID) {
389  if (!graphLoaded) {
390  KATANA_DIE("Graph hasn't been loaded yet.");
391  }
392 
393  if (numLocalNodes == 0) {
394  return EdgeIterator(0);
395  }
396  KATANA_LOG_DEBUG_ASSERT(nodeOffset <= globalNodeID);
397  KATANA_LOG_DEBUG_ASSERT(globalNodeID < (nodeOffset + numLocalNodes));
398 
399  uint64_t localNodeID = globalNodeID - nodeOffset;
400 
401  if (localNodeID != 0) {
402  numBytesReadOutIndex += sizeof(uint64_t);
403  return EdgeIterator(outIndexBuffer[localNodeID - 1]);
404  } else {
405  return EdgeIterator(edgeOffset);

407  }
