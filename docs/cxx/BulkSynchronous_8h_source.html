89  std::optional<value_type> pop() {
90  TLD& tld = *tlds.getLocal();
91  std::optional<value_type> r;
92 
93  while (true) {
94  if (isEmpty)
95  return r; // empty
96 
97  r = wls[tld.round].pop();
98  if (r)
99  return r;
100 
101  barrier.Wait();
102  if (ThreadPool::getTID() == 0) {
103  if (!some.get())
104  isEmpty = true;
105  some.get() = false;
106  }
107  tld.round = (tld.round + 1) & 1;
108  barrier.Wait();
109 
110  r = wls[tld.round].pop();
111  if (r) {
112  some.get() = true;
113  return r;
114  }
115  }
116  }
