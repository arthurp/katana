110  Chunk* popChunk() {
111  int id = Q.myEffectiveID();
112  Chunk* r = popChunkByID(id);
113  if (r)
114  return r;
115 
116  for (int i = id + 1; i < (int)Q.size(); ++i) {
117  r = popChunkByID(i);
118  if (r)
119  return r;
120  }
121 
122  for (int i = 0; i < id; ++i) {
123  r = popChunkByID(i);
124  if (r)
125  return r;
126  }
127 
128  return 0;
129  }

131  template <typename... Args>
132  T* emplacei(p& n, Args&&... args) {
133  T* retval = 0;
134  if (n.next && (retval = n.next->emplace_back(std::forward<Args>(args)...)))
135  return retval;
136  if (n.next)
137  pushChunk(n.next);
138  n.next = mkChunk();
139  retval = n.next->emplace_back(std::forward<Args>(args)...);
140  KATANA_LOG_DEBUG_ASSERT(retval);
141  return retval;
142  }

176  value_type* peek() {
177  p& n = data.get();
178  if (IsStack) {
179  if (n.next && !n.next->empty())
180  return &n.next->back();
181  if (n.next)
182  delChunk(n.next);
183  n.next = popChunk();
184  if (n.next && !n.next->empty())
185  return &n.next->back();
186  return NULL;
187  } else {
188  if (n.cur && !n.cur->empty())
189  return &n.cur->front();
190  if (n.cur)
191  delChunk(n.cur);
192  n.cur = popChunk();
193  if (!n.cur) {
194  n.cur = n.next;
195  n.next = 0;
196  }
197  if (n.cur && !n.cur->empty())
198  return &n.cur->front();
199  return NULL;
200  }
201  }

236  std::optional<value_type> pop() {
237  p& n = data.get();
238  std::optional<value_type> retval;
239  if (IsStack) {
240  if (n.next && (retval = n.next->extract_back()))
241  return retval;
242  if (n.next)
243  delChunk(n.next);
244  n.next = popChunk();
245  if (n.next)
246  return n.next->extract_back();
247  return std::nullopt;
248  } else {
249  if (n.cur && (retval = n.cur->extract_front()))
250  return retval;
251  if (n.cur)
252  delChunk(n.cur);
253  n.cur = popChunk();
254  if (!n.cur) {
255  n.cur = n.next;
256  n.next = 0;
257  }
258  if (n.cur)
259  return n.cur->extract_front();
260  return std::nullopt;
261  }
