71  template <typename EdgeWeightType>
72  void FindNeighboringClusters(
73  const Graph& graph, GNode& n,
74  std::map<uint64_t, uint64_t>& cluster_local_map,
75  std::vector<EdgeTy>& counter, EdgeTy& self_loop_wt) {
76  uint64_t num_unique_clusters = 0;
77 
78  // Add the node's current cluster to be considered
79  // for movement as well
80  cluster_local_map[graph.template GetData<CurrentCommunityId>(n)] =
81  0; // Add n's current cluster
82  counter.push_back(0); // Initialize the counter to zero (no edges incident
83  // yet)
84  num_unique_clusters++;
85 
86  // Assuming we have grabbed lock on all the neighbors
87  for (auto ii = graph.edge_begin(n); ii != graph.edge_end(n); ++ii) {
88  auto dst = graph.GetEdgeDest(ii);
89  auto edge_wt = graph.template GetEdgeData<EdgeWeight<EdgeWeightType>>(
90  ii); // Self loop weights is recorded
91  if (*dst == n) {
92  self_loop_wt += edge_wt; // Self loop weights is recorded
93  }
94  auto stored_already =
95  cluster_local_map.find(graph.template GetData<CurrentCommunityId>(
96  dst)); // Check if it already exists
97  if (stored_already != cluster_local_map.end()) {
98  counter[stored_already->second] += edge_wt;
99  } else {
100  cluster_local_map[graph.template GetData<CurrentCommunityId>(dst)] =
101  num_unique_clusters;
102  counter.push_back(edge_wt);
103  num_unique_clusters++;
104  }
105  } // End edge loop
106  return;
107  }

114  uint64_t VertexFollowing(Graph* graph) {
115  using GNode = typename Graph::Node;
116  // Initialize each node to its own cluster
117  katana::do_all(katana::iterate(*graph), [&](GNode n) {
118  graph->template GetData<CurrentCommunityId>(n) = n;
119  });
120 
121  // Remove isolated and degree-one nodes
122  katana::GAccumulator<uint64_t> isolated_nodes;
123  katana::do_all(katana::iterate(*graph), [&](GNode n) {
124  auto& n_data_curr_comm_id =
125  graph->template GetData<CurrentCommunityId>(n);
126  uint64_t degree = std::distance(graph->edge_begin(n), graph->edge_end(n));
127  if (degree == 0) {
128  isolated_nodes += 1;
129  n_data_curr_comm_id = UNASSIGNED;
130  } else {
131  if (degree == 1) {
132  // Check if the destination has degree greater than one
133  auto dst = graph->GetEdgeDest(graph->edge_end(n) - 1);
134  uint64_t dst_degree =
135  std::distance(graph->edge_begin(*dst), graph->edge_end(*dst));
136  if ((dst_degree > 1 || (n > *dst))) {
137  isolated_nodes += 1;
138  n_data_curr_comm_id =
139  graph->template GetData<CurrentCommunityId>(dst);
140  }
141  }
142  }
143  });
144  // The number of isolated nodes that can be removed
145  return isolated_nodes.reduce();
146  }

209  uint64_t MaxModularityWithoutSwaps(
210  std::map<uint64_t, uint64_t>& cluster_local_map,
211  std::vector<EdgeTy>& counter, uint64_t self_loop_wt,
212  CommunityArray& c_info, EdgeTy degree_wt, uint64_t sc, double constant) {
213  uint64_t max_index = sc; // Assign the intial value as self community
214  double cur_gain = 0;
215  double max_gain = 0;
216  double eix = counter[0] - self_loop_wt;
217  double ax = c_info[sc].degree_wt - degree_wt;
218  double eiy = 0;
219  double ay = 0;
220 
221  auto stored_already = cluster_local_map.begin();
222  do {
223  if (sc != stored_already->first) {
224  ay = c_info[stored_already->first].degree_wt; // Degree wt of cluster y
225 
226  if (ay < (ax + degree_wt)) {
227  stored_already++;
228  continue;
229  } else if (ay == (ax + degree_wt) && stored_already->first > sc) {
230  stored_already++;
231  continue;
232  }
233 
234  eiy = counter[stored_already
235  ->second]; // Total edges incident on cluster y
236  cur_gain = 2 * constant * (eiy - eix) +
237  2 * degree_wt * ((ax - ay) * constant * constant);
238 
239  if ((cur_gain > max_gain) ||
240  ((cur_gain == max_gain) && (cur_gain != 0) &&
241  (stored_already->first < max_index))) {
242  max_gain = cur_gain;
243  max_index = stored_already->first;
244  }
245  }
246  stored_already++; // Explore next cluster
247  } while (stored_already != cluster_local_map.end());
248 
249  if ((c_info[max_index].size == 1 && c_info[sc].size == 1 &&
250  max_index > sc)) {
251  max_index = sc;
252  }
253 
254  KATANA_LOG_DEBUG_ASSERT(max_gain >= 0);
255  return max_index;
256  }

261  template <typename EdgeWeightType>
262  double CalModularity(
263  const Graph& graph, CommunityArray& c_info, double& e_xx, double& a2_x,
264  double& constant_for_second_term) {
265  /* Variables needed for Modularity calculation */
266  double mod = -1;
267 
268  katana::LargeArray<EdgeTy> cluster_wt_internal;
269 
270  /*** Initialization ***/
271  cluster_wt_internal.allocateBlocked(graph.num_nodes());
272 
273  /* Calculate the overall modularity */
274  katana::GAccumulator<double> acc_e_xx;
275  katana::GAccumulator<double> acc_a2_x;
276 
277  katana::do_all(
278  katana::iterate(graph), [&](GNode n) { cluster_wt_internal[n] = 0; });
279 
280  katana::do_all(katana::iterate(graph), [&](GNode n) {
281  auto n_data_current_comm_id =
282  graph.template GetData<CurrentCommunityId>(n);
283  for (auto ii = graph.edge_begin(n); ii != graph.edge_end(n); ++ii) {
284  if (graph.template GetData<CurrentCommunityId>(graph.GetEdgeDest(ii)) ==
285  n_data_current_comm_id) {
286  cluster_wt_internal[n] +=
287  graph.template GetEdgeData<EdgeWeight<EdgeWeightType>>(ii);
288  }
289  }
290  });
291 
292  katana::do_all(katana::iterate(graph), [&](GNode n) {
293  acc_e_xx += cluster_wt_internal[n];
294  acc_a2_x +=
295  (double)(c_info[n].degree_wt) *
296  ((double)(c_info[n].degree_wt) * (double)constant_for_second_term);
297  });
298 
299  e_xx = acc_e_xx.reduce();
300  a2_x = acc_a2_x.reduce();
301 
302  mod = e_xx * (double)constant_for_second_term -
303  a2_x * (double)constant_for_second_term;
304  return mod;
305  }

335  template <typename GraphTy, typename EdgeWeightType>
336  static double CalModularityFinal(GraphTy& graph) {
337  CommunityArray c_info; // Community info
338  CommunityArray c_update; // Used for updating community
339 
340  /* Variables needed for Modularity calculation */
341  double constant_for_second_term;
342  double mod = -1;
343 
344  katana::LargeArray<EdgeTy> cluster_wt_internal;
345 
346  /*** Initialization ***/
347  c_info.allocateBlocked(graph.num_nodes());
348  c_update.allocateBlocked(graph.num_nodes());
349  cluster_wt_internal.allocateBlocked(graph.num_nodes());
350 
351  katana::LargeArray<EdgeWeightType> degree_weight_array;
352  degree_weight_array.allocateBlocked(graph.num_nodes());
353 
354  /* Calculate the weighted degree sum for each vertex */
355  SumClusterWeight<EdgeWeightType>(graph, c_info, degree_weight_array);
356 
357  /* Compute the total weight (2m) and 1/2m terms */
358  constant_for_second_term =
359  CalConstantForSecondTerm<EdgeWeightType>(graph, degree_weight_array);
360 
361  /* Calculate the overall modularity */
362  double e_xx = 0;
363  katana::GAccumulator<double> acc_e_xx;
364  double a2_x = 0;
365  katana::GAccumulator<double> acc_a2_x;
366 
367  katana::do_all(
368  katana::iterate(graph), [&](GNode n) { cluster_wt_internal[n] = 0; });
369 
370  katana::do_all(katana::iterate(graph), [&](GNode n) {
371  auto n_data_current_comm = graph.template GetData<CurrentCommunityId>(n);
372  for (auto ii = graph.edge_begin(n); ii != graph.edge_end(n); ++ii) {
373  if (graph.template GetData<CurrentCommunityId>(graph.GetEdgeDest(ii)) ==
374  n_data_current_comm) {
375  // if(graph.getData(graph.getEdgeDst(ii)).prev_comm_ass ==
376  // n_data.prev_comm_ass) {
377  cluster_wt_internal[n] +=
378  graph.template GetEdgeData<EdgeWeight<EdgeWeightType>>(ii);
379  }
380  }
381  });
382 
383  katana::do_all(katana::iterate(graph), [&](GNode n) {
384  acc_e_xx += cluster_wt_internal[n];
385  acc_a2_x +=
386  (double)(c_info[n].degree_wt) *
387  ((double)(c_info[n].degree_wt) * (double)constant_for_second_term);
388  });
389 
390  e_xx = acc_e_xx.reduce();
391  a2_x = acc_a2_x.reduce();
392 
393  mod = e_xx * (double)constant_for_second_term -
394  a2_x * (double)constant_for_second_term;
395  return mod;
396  }

443  katana::Result<void> CreateDuplicateGraph(
444  katana::PropertyGraph* pfg_from, katana::PropertyGraph* pfg_to,
445  const std::string& edge_property_name,
446  const std::string& new_edge_property_name) {

454  auto out_indices_view_from_result =
455  katana::ConstructPropertyView<katana::UInt64Property>(
456  topology_from.out_indices.get());
457  if (!out_indices_view_from_result) {
458  return out_indices_view_from_result.error();
459  }
460  auto out_indices_view_from =
461  std::move(out_indices_view_from_result.value());

465  topology_from.out_dests.get());
466  if (!out_dests_view_from_result) {
467  return out_dests_view_from_result.error();
468  }
469  auto out_dests_view_from = std::move(out_dests_view_from_result.value());
470 
471  auto out_indices_view_to_result =
472  katana::ConstructPropertyView<katana::UInt64Property>(
473  topology_to.out_indices.get());
474  if (!out_indices_view_to_result) {
475  return out_indices_view_to_result.error();
476  }
477  auto out_indices_view_to = std::move(out_indices_view_to_result.value());
478 
479  auto out_dests_view_to_result =
480  katana::ConstructPropertyView<katana::UInt32Property>(
481  topology_to.out_dests.get());
482  if (!out_dests_view_to_result) {
483  return out_dests_view_to_result.error();
484  }
485  auto out_dests_view_to = std::move(out_dests_view_to_result.value());
486 
487  // First pass to find the number of edges
488  katana::do_all(
489  katana::iterate((uint64_t)0, topology_from.num_nodes()),
490  [&](uint64_t n) { out_indices_view_to[n] = out_indices_view_from[n]; });
491  katana::do_all(
492  katana::iterate((uint64_t)0, topology_from.num_edges()),
493  [&](uint64_t e) { out_dests_view_to[e] = out_dests_view_from[e]; });
494 
495  // Remove the existing edge property
496  if (auto r = pfg_to->RemoveEdgeProperty(new_edge_property_name); !r) {
497  return r.error();
498  }
499  // Copy edge properties
500  using ArrowType = typename arrow::CTypeTraits<EdgeTy>::ArrowType;
501  auto edge_property_result =
502  pfg_from->GetEdgePropertyTyped<EdgeTy>(edge_property_name);
503  if (!edge_property_result) {
504  return edge_property_result.error();
505  }
506  auto edge_property = edge_property_result.value();
507  std::vector<std::shared_ptr<arrow::Field>> fields;
508  std::vector<std::shared_ptr<arrow::Array>> columns;
509  fields.emplace_back(
510  arrow::field((new_edge_property_name), std::make_shared<ArrowType>()));
511  columns.emplace_back(edge_property);
512  auto edge_data_table = arrow::Table::Make(arrow::schema(fields), columns);
513  if (auto r = pfg_to->AddEdgeProperties(edge_data_table); !r) {
514  return r;
515  }
516  return katana::ResultSuccess();
517  }

528  template <typename NodeData, typename EdgeData, typename EdgeWeightType>
529  katana::Result<std::unique_ptr<katana::PropertyGraph>> GraphCoarsening(
530  const Graph& graph, katana::PropertyGraph* pfg_mutable,
531  uint64_t num_unique_clusters,
532  const std::vector<std::string>& temp_node_property_names,
533  const std::vector<std::string>& temp_edge_property_names) {
534  using GNode = typename Graph::Node;
535 
536  katana::StatTimer TimerGraphBuild("Timer_Graph_build");
537  TimerGraphBuild.start();
538  uint64_t num_nodes_next = num_unique_clusters;
539  uint64_t num_edges_next = 0; // Unknown right now
540 
541  std::vector<std::vector<GNode>> cluster_bags(num_unique_clusters);
542  // Comment: Serial separation is better than do_all due to contention
543  for (GNode n = 0; n < graph.num_nodes(); ++n) {
544  auto n_data_curr_comm_id = graph.template GetData<CurrentCommunityId>(n);
545  if (n_data_curr_comm_id != UNASSIGNED)
546  cluster_bags[n_data_curr_comm_id].push_back(n);
547  }
548 
549  std::vector<std::vector<uint32_t>> edges_id(num_unique_clusters);
550  std::vector<std::vector<EdgeTy>> edges_data(num_unique_clusters);
551 
552  /* First pass to find the number of edges */
553  katana::do_all(
554  katana::iterate((uint64_t)0, num_unique_clusters),
555  [&](uint64_t c) {
556  std::map<uint64_t, uint64_t> cluster_local_map;
557  uint64_t num_unique_clusters = 0;
558  for (auto cb_ii = cluster_bags[c].begin();
559  cb_ii != cluster_bags[c].end(); ++cb_ii) {
560  KATANA_LOG_DEBUG_ASSERT(
561  graph.template GetData<CurrentCommunityId>(*cb_ii) ==
562  c); // All nodes in this bag must have same cluster id
563 
564  for (auto ii = graph.edge_begin(*cb_ii);
565  ii != graph.edge_end(*cb_ii); ++ii) {
566  auto dst = graph.GetEdgeDest(ii);
567  auto dst_data_curr_comm_id =
568  graph.template GetData<CurrentCommunityId>(dst);
569  KATANA_LOG_DEBUG_ASSERT(dst_data_curr_comm_id != UNASSIGNED);
570  auto stored_already = cluster_local_map.find(
571  dst_data_curr_comm_id); // Check if it already exists
572  if (stored_already != cluster_local_map.end()) {
573  edges_data[c][stored_already->second] +=
574  graph.template GetEdgeData<EdgeWeight<EdgeWeightType>>(ii);
575  } else {
576  cluster_local_map[dst_data_curr_comm_id] = num_unique_clusters;
577  edges_id[c].push_back(dst_data_curr_comm_id);
578  edges_data[c].push_back(
579  graph.template GetEdgeData<EdgeWeight<EdgeWeightType>>(ii));
580  num_unique_clusters++;
581  }
582  } // End edge loop
583  }
584  },
585  katana::steal(), katana::loopname("BuildGraph: Find edges"));
586 
587  /* Serial loop to reduce all the edge counts */
588  std::vector<uint64_t> prefix_edges_count(num_unique_clusters);
589  katana::GAccumulator<uint64_t> num_edges_acc;
590  katana::do_all(
591  katana::iterate((uint64_t)0, num_nodes_next), [&](uint64_t c) {
592  prefix_edges_count[c] = edges_id[c].size();
593  num_edges_acc += prefix_edges_count[c];
594  });
595 
596  num_edges_next = num_edges_acc.reduce();
597  for (uint64_t c = 1; c < num_nodes_next; ++c) {
598  prefix_edges_count[c] += prefix_edges_count[c - 1];
599  }
600 
601  KATANA_LOG_DEBUG_ASSERT(
602  prefix_edges_count[num_unique_clusters - 1] == num_edges_next);
603  katana::StatTimer TimerConstructFrom("Timer_Construct_From");
604  TimerConstructFrom.start();
605 
606  // Remove all the existing node/edge properties
607  for (auto property : temp_node_property_names) {
608  if (auto r = pfg_mutable->RemoveNodeProperty(property); !r) {
609  return r.error();
610  }
611  }
612  for (auto property : temp_edge_property_names) {
613  if (auto r = pfg_mutable->RemoveEdgeProperty(property); !r) {
614  return r.error();
615  }
616  }
617 
618  const katana::GraphTopology& topology = pfg_mutable->topology();
619  auto out_indices_next =
620  (topology.out_indices.get())
621  ->Slice(0, static_cast<int64_t>(num_nodes_next));
622  std::shared_ptr<arrow::Array> out_dests_next =
623  (topology.out_dests.get())
