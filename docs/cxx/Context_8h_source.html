148  void addToNhood(Lockable* lockable) {
149  KATANA_LOG_DEBUG_ASSERT(!lockable->next);
150  lockable->next = locks;
151  locks = lockable;
152  }
153 
154  void acquire(Lockable* lockable, katana::MethodFlag m) {
155  AcquireStatus i;
156  if (customAcquire) {
157  subAcquire(lockable, m);
158  } else if ((i = tryAcquire(lockable)) != AcquireStatus::FAIL) {
159  if (i == AcquireStatus::NEW_OWNER) {
160  addToNhood(lockable);
161  }
162  } else {
163  signalConflict(lockable);
164  }
165  }
166 
167  void release(Lockable* lockable);

169 public:
170  SimpleRuntimeContext(bool child = false) : locks(0), customAcquire(child) {}
171  virtual ~SimpleRuntimeContext() {}

187 inline bool
188 shouldLock(const katana::MethodFlag g) {
189  // Mask out additional "optional" flags
190  switch (g & katana::MethodFlag::INTERNAL_MASK) {
191  case MethodFlag::UNPROTECTED:
192  case MethodFlag::PREVIOUS:
193  return false;
194 
195  case MethodFlag::READ:
196  case MethodFlag::WRITE:
197  return true;
198 
199  default:
200  // Adding error checking code here either upsets the inlining heuristics or
201  // icache behavior. Avoid complex code if possible.
202  assert(false);
203  }
204  return false;
205 }

209 doAcquire(Lockable* lockable, katana::MethodFlag m) {
