36 template <typename FunctionTy, typename ArgsTy>
37 inline void
38 on_each_impl(FunctionTy&& fn, const ArgsTy& argsTuple) {
39  static_assert(!has_trait<char*, ArgsTy>(), "old loopname");
40  static_assert(!has_trait<char const*, ArgsTy>(), "old loopname");
41 
42  static constexpr bool NEEDS_STATS = has_trait<loopname_tag, ArgsTy>();
43  static constexpr bool MORE_STATS =
44  NEEDS_STATS && has_trait<more_stats_tag, ArgsTy>();
45 
46  const char* const loopname = katana::internal::getLoopName(argsTuple);
47 
48  CondStatTimer<NEEDS_STATS> timer(loopname);
49 
50  PerThreadTimer<MORE_STATS> execTime(loopname, "Execute");
51 
52  const auto numT = getActiveThreads();
53 
54  OperatorReferenceType<decltype(std::forward<FunctionTy>(fn))> fn_ref = fn;
55 
56  auto runFun = [&] {
57  execTime.start();
58 
59  fn_ref(ThreadPool::getTID(), numT);
60 
61  execTime.stop();
62  };
63 
64  timer.start();
65  GetThreadPool().run(numT, runFun);
66  timer.stop();
67 }
68 
69 } // namespace internal
