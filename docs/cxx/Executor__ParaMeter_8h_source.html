112 template <typename T>
113 class FIFO_WL {
114  using PTcont = katana::PerThreadStorage<katana::gstl::Vector<T>>;
115 
116  std::array<PTcont, 2> worklists;
117 
118  PTcont* curr;
119  PTcont* next;
120 
121 public:
122  FIFO_WL() : curr(&worklists[0]), next(&worklists[1]) {}
123 
124  auto iterateCurr() { return katana::MakeLocalTwoLevelRange(*curr); }
125 
126  void pushNext(const T& item) { next->getLocal()->push_back(item); }
127 
128  void nextStep() {
129  std::swap(curr, next);
130  katana::on_each_gen(
131  [this](const unsigned, const unsigned) { next->getLocal()->clear(); },
132  std::make_tuple());
133  }
134 
135  PTcont* currentWorklist() { return curr; }
136 
137  bool empty() const {
138  for (unsigned i = 0, n = next->size(); i < n; ++i) {
139  if (!next->getRemote(i)->empty()) {
140  return false;
141  }
142  }
143  return true;
144  }
145 };

147 template <typename T>
148 class RAND_WL : public FIFO_WL<T> {
149 public:
150  auto iterateCurr() {
151  katana::on_each_gen(
152  [&](int, int) {
153  auto& lwl = *this->currentWorklist()->getLocal();
154 
155  std::random_device r;
156  std::mt19937 rng(r());
157  std::shuffle(lwl.begin(), lwl.end(), rng);
158  },
159  std::make_tuple());
160 
161  return MakeLocalTwoLevelRange(*this->currentWorklist());
162  }

202 class ParaMeterExecutor {
203  using value_type = T;
204  using GenericWL = typename trait_type<wl_tag, ArgsTy>::type::type;
205  using WorkListTy = typename GenericWL::template retype<T>;
206  using dbg = katana::debug<1>;
207 
208  constexpr static bool needsStats = !has_trait<no_stats_tag, ArgsTy>();
209  constexpr static bool needsPush = !has_trait<no_pushes_tag, ArgsTy>();
210  constexpr static bool needsAborts =
211  !has_trait<disable_conflict_detection_tag, ArgsTy>();
212  constexpr static bool needsPia = has_trait<per_iter_alloc_tag, ArgsTy>();
213  constexpr static bool needsBreak = has_trait<parallel_break_tag, ArgsTy>();

215  struct IterationContext {
216  T item;
217  bool doabort;
218  katana::UserContextAccess<value_type> facing;
219  SimpleRuntimeContext ctx;
220 
221  explicit IterationContext(const T& v) : item(v), doabort(false) {}
222 
223  void reset() {
224  doabort = false;
225  if (needsPia)
226  facing.resetAlloc();
227 
228  if (needsPush)
229  facing.getPushBuffer().clear();
230  }
231  };
232 
233  using PWL = typename ChooseWL<IterationContext*, WorkListTy::SCHEDULE>::type;
234 
235 private:
236  PWL m_wl;
237  FunctionTy m_func;
238  const char* loopname;
239  FILE* m_statsFile;
240  FixedSizeAllocator<IterationContext> m_iterAlloc;
241  katana::GReduceLogicalOr m_broken;
242 
243  IterationContext* newIteration(const T& item) {
244  IterationContext* it = m_iterAlloc.allocate(1);
245  KATANA_LOG_DEBUG_VASSERT(it, "IterationContext allocation failed");
246 
247  m_iterAlloc.construct(it, item);
248 
249  it->reset();
250  return it;
251  }
252 
253  unsigned abortIteration(IterationContext* it) {
254  KATANA_LOG_DEBUG_VASSERT(it, "nullptr arg");
255  assert(
256  it->doabort &&
257  "aborting an iteration without setting its doabort flag");
258 
259  unsigned numLocks = it->ctx.cancelIteration();
260  it->reset();
261 
262  m_wl.pushNext(it);
263  return numLocks;
264  }
265 
266  unsigned commitIteration(IterationContext* it) {
267  KATANA_LOG_DEBUG_VASSERT(it, "nullptr arg");
268 
269  if (needsPush) {
270  for (const auto& item : it->facing.getPushBuffer()) {
271  IterationContext* child = newIteration(item);
272  m_wl.pushNext(child);
273  }
274  }
275 
276  unsigned numLocks = it->ctx.commitIteration();
277  it->reset();
278 
279  m_iterAlloc.destroy(it);
280  m_iterAlloc.deallocate(it, 1);
281 
282  return numLocks;
283  }

285 private:
286  void runSimpleStep(UnorderedStepStats& stats) {
287  katana::do_all_gen(
288  m_wl.iterateCurr(),
289  [&, this](IterationContext* it) {
290  stats.wlSize += 1;
291 
292  setThreadContext(&(it->ctx));
293 
294  m_func(it->item, it->facing.data());
295  stats.parallelism += 1;
296  unsigned nh = commitIteration(it);
297  stats.nhSize += nh;
298 
299  setThreadContext(nullptr);
300  },
301  std::make_tuple(katana::steal(), katana::loopname("ParaM-Simple")));
302  }
303 
304  void runCautiousStep(UnorderedStepStats& stats) {
305  katana::do_all_gen(
306  m_wl.iterateCurr(),
307  [&, this](IterationContext* it) {
308  stats.wlSize += 1;
309 
310  setThreadContext(&(it->ctx));
311  bool broke = false;
312 
313  if (needsBreak) {
314  it->facing.setBreakFlag(&broke);
315  }
316 #ifdef KATANA_USE_LONGJMP_ABORT
317  int flag = 0;
318  if ((flag = setjmp(execFrame)) == 0) {
319  m_func(it->item, it->facing.data());
320  } else
321 #elif KATANA_USE_EXCEPTION_ABORT
322  try {
323  m_func(it->item, it->facing.data());
324 
325  } catch (const ConflictFlag& flag)
326 #endif
327  {
328  clearConflictLock();
329  switch (flag) {
330  case katana::CONFLICT:
331  it->doabort = true;
332  break;
333  default:
334  std::abort();
335  }
336  }
337 
338  if (needsBreak && broke) {
339  m_broken.update(true);
340  }
341 
342  setThreadContext(nullptr);
343  },
344  std::make_tuple(katana::steal(), katana::loopname("ParaM-Expand-NH")));
345 
346  katana::do_all_gen(
347  m_wl.iterateCurr(),
348  [&, this](IterationContext* it) {
349  if (it->doabort) {
350  abortIteration(it);
351 
352  } else {
353  stats.parallelism += 1;
354  unsigned nh = commitIteration(it);
355  stats.nhSize += nh;
356  }
357  },
358  std::make_tuple(katana::steal(), katana::loopname("ParaM-Commit")));
359  }

361  template <typename R>
362  void execute(const R& range) {
363  katana::on_each_gen(
364  [&, this](const unsigned, const unsigned) {
365  auto begin = range.local_begin();
366  auto end = range.local_end();
367 
368  for (auto i = begin; i != end; ++i) {
369  IterationContext* it = newIteration(*i);
370  m_wl.pushNext(it);
371  }
372  },
373  std::make_tuple());
374 
375  UnorderedStepStats stats;
376 
377  while (!m_wl.empty()) {
378  m_wl.nextStep();
379 
380  if (needsAborts) {
381  runCautiousStep(stats);
382 
383  } else {
384  runSimpleStep(stats);
385  }
386 
387  // dbg::print("Step: ", stats.step, ", Parallelism: ",
388  // stats.parallelism.reduce());
389  KATANA_LOG_DEBUG_VASSERT(
390  stats.parallelism.reduce(), "ERROR: No Progress");
391 
392  stats.dump(m_statsFile, loopname);
393  stats.nextStep();
394 
395  if (needsBreak && m_broken.reduce()) {
396  break;
397  }
398 
399  } // end while
400 
401  closeStatsFile();
402  }
