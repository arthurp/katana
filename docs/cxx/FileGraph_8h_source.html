680 class FileGraphWriter : public FileGraph {
681  std::vector<uint64_t> outIdx;
682  std::vector<uint32_t> starts;
683  std::vector<uint32_t> outs;
684  std::vector<uint64_t> starts64;
685  std::vector<uint64_t> outs64;
686 
687  size_t sizeofEdgeData;
688  size_t numNodes;
689  size_t numEdges;
690 
691 public:
693  FileGraphWriter() : sizeofEdgeData(0), numNodes(0), numEdges(0) {}
694 
697  void setNumNodes(size_t n) { numNodes = n; }
700  void setNumEdges(size_t n) { numEdges = n; }
703  void setSizeofEdgeData(size_t n) { sizeofEdgeData = n; }
704 
707  void phase1() { outIdx.resize(numNodes); }
708 
710  void incrementDegree(size_t id, int delta = 1) {
711  KATANA_LOG_DEBUG_ASSERT(id < numNodes);
712  outIdx[id] += delta;
713  }
714 
716  void phase2() {
717  if (numNodes == 0)
718  return;
719 
720  // Turn counts into partial sums
721  auto prev = outIdx.begin();
722  for (auto ii = outIdx.begin() + 1, ei = outIdx.end(); ii != ei;
723  ++ii, ++prev) {
724  *ii += *prev;
725  }
726  KATANA_LOG_DEBUG_ASSERT(outIdx[numNodes - 1] == numEdges);
727 
728  if (numNodes <= std::numeric_limits<uint32_t>::max()) {
729  // version 1
730  starts.resize(numNodes);
731  outs.resize(numEdges);
732  } else {
733  // version 2
734  starts64.resize(numNodes);
735  outs64.resize(numEdges);
736  }
737  }
738 
740  size_t addNeighbor(size_t src, size_t dst) {
741  size_t base = src ? outIdx[src - 1] : 0;
742 
743  if (numNodes <= std::numeric_limits<uint32_t>::max()) {
744  // version 1
745  size_t idx = base + starts[src]++;
746  KATANA_LOG_DEBUG_ASSERT(idx < outIdx[src]);
747  outs[idx] = dst;
748  return idx;
749  } else {
750  // version 2
751  size_t idx = base + (starts64)[src]++;
752  KATANA_LOG_DEBUG_ASSERT(idx < outIdx[src]);
753  outs64[idx] = dst;
754  return idx;
755  }
756  }
757 
762  template <typename T>
763  T* finish() {
764  void* ret;
765  if (numNodes <= std::numeric_limits<uint32_t>::max()) {
766  // version 1
767  ret = fromArrays(
768  &outIdx[0], numNodes, &outs[0], numEdges, nullptr, sizeofEdgeData, 0,
769  0, false, 1);
770  starts.clear();
771  outs.clear();
772  } else {
773  // version 2
774  ret = fromArrays(
775  &outIdx[0], numNodes, &outs64[0], numEdges, nullptr, sizeofEdgeData,
776  0, 0, false, 2);
777  }
778 
779  outIdx.clear();
780  return reinterpret_cast<T*>(ret);
781  }
782 };

789 template <typename EdgeTy>
790 void
791 makeSymmetric(FileGraph& in_graph, FileGraph& out) {
792  typedef FileGraph::GraphNode GNode;
793  typedef LargeArray<EdgeTy> EdgeData;
794  typedef typename EdgeData::value_type edge_value_type;
795 
796  FileGraphWriter g;
797  EdgeData edgeData;
798 
799  size_t numEdges = 0;
800 
801  for (FileGraph::iterator ii = in_graph.begin(), ei = in_graph.end(); ii != ei;
802  ++ii) {
803  GNode src = *ii;
804  for (FileGraph::edge_iterator jj = in_graph.edge_begin(src),
805  ej = in_graph.edge_end(src);
806  jj != ej; ++jj) {
807  GNode dst = in_graph.getEdgeDst(jj);
808  numEdges += 1;
809  if (src != dst)
810  numEdges += 1;
811  }
812  }
813 
814  g.setNumNodes(in_graph.size());
815  g.setNumEdges(numEdges);
816  g.setSizeofEdgeData(EdgeData::size_of::value);
817 
818  g.phase1();
819  for (FileGraph::iterator ii = in_graph.begin(), ei = in_graph.end(); ii != ei;
820  ++ii) {
821  GNode src = *ii;
822  for (FileGraph::edge_iterator jj = in_graph.edge_begin(src),
823  ej = in_graph.edge_end(src);
824  jj != ej; ++jj) {
825  GNode dst = in_graph.getEdgeDst(jj);
826  g.incrementDegree(src);
827  if (src != dst)
828  g.incrementDegree(dst);
829  }
830  }
831 
832  g.phase2();
833  edgeData.create(numEdges);
834  for (FileGraph::iterator ii = in_graph.begin(), ei = in_graph.end(); ii != ei;
835  ++ii) {
836  GNode src = *ii;
837  for (FileGraph::edge_iterator jj = in_graph.edge_begin(src),
838  ej = in_graph.edge_end(src);
839  jj != ej; ++jj) {
840  GNode dst = in_graph.getEdgeDst(jj);
841  if (EdgeData::has_value) {
842  edge_value_type& data = in_graph.getEdgeData<edge_value_type>(jj);
843  edgeData.set(g.addNeighbor(src, dst), data);
844  if (src != dst)
845  edgeData.set(g.addNeighbor(dst, src), data);
846  } else {
847  g.addNeighbor(src, dst);
848  if (src != dst)
849  g.addNeighbor(dst, src);
850  }
851  }
852  }
853 
854  edge_value_type* rawEdgeData = g.finish<edge_value_type>();
855  if (EdgeData::has_value)
856  std::uninitialized_copy(
857  std::make_move_iterator(edgeData.begin()),
858  std::make_move_iterator(edgeData.end()), rawEdgeData);
859 
860  out = std::move(g);
861 }

874 template <typename EdgeTy, typename PTy>
875 void
876 permute(FileGraph& in_graph, const PTy& p, FileGraph& out) {
877  typedef FileGraph::GraphNode GNode;
878  typedef LargeArray<EdgeTy> EdgeData;
879  typedef typename EdgeData::value_type edge_value_type;
880 
881  FileGraphWriter g;
882  EdgeData edgeData;
883 
884  size_t numEdges = in_graph.sizeEdges();
885  g.setNumNodes(in_graph.size());
886  g.setNumEdges(numEdges);
887  g.setSizeofEdgeData(EdgeData::size_of::value);
888 
889  g.phase1();
890  for (FileGraph::iterator ii = in_graph.begin(), ei = in_graph.end(); ii != ei;
891  ++ii) {
892  GNode src = *ii;
893  for (FileGraph::edge_iterator jj = in_graph.edge_begin(src),
894  ej = in_graph.edge_end(src);
895  jj != ej; ++jj) {
896  g.incrementDegree(p[src]);
897  }
898  }
899 
900  g.phase2();
901  edgeData.create(numEdges);
902  for (FileGraph::iterator ii = in_graph.begin(), ei = in_graph.end(); ii != ei;
903  ++ii) {
904  GNode src = *ii;
905  for (FileGraph::edge_iterator jj = in_graph.edge_begin(src),
906  ej = in_graph.edge_end(src);
907  jj != ej; ++jj) {
908  GNode dst = in_graph.getEdgeDst(jj);
909  if (EdgeData::has_value) {
910  edge_value_type& data = in_graph.getEdgeData<edge_value_type>(jj);
911  edgeData.set(g.addNeighbor(p[src], p[dst]), data);
912  } else {
913  g.addNeighbor(p[src], p[dst]);
914  }
915  }
916  }
917 
918  edge_value_type* rawEdgeData = g.finish<edge_value_type>();
919  if (EdgeData::has_value)
920  std::uninitialized_copy(
921  std::make_move_iterator(edgeData.begin()),
922  std::make_move_iterator(edgeData.end()), rawEdgeData);
923 
924  out = std::move(g);
