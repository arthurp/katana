177  template <typename... Args>
178  std::pair<iterator, bool> emplace(Args&&... args) {
179  // KATANA_LOG_DEBUG_ASSERT(std::adjacent_find(_data.begin(), _data.end(), [&](const
180  // value_type& a, const value_type& b) {
181  // return key_comp()(b.first, a.first);
182  //}) == _data.end());
183  _data.emplace_back(std::forward<Args>(args)...);
184  value_type& v = _data.back();
185  auto ee = _data.end();
186  --ee;
187  auto __i = std::lower_bound(_data.begin(), ee, v.first, value_key_comp());
188  // key < __i->first
189  bool retval = __i == ee || key_comp()(v.first, (*__i).first);
190  if (retval) {
191  if (__i != ee) {
192  value_type tmp = std::move(v);
193  __i = _data.emplace(__i, std::move(tmp));
194  _data.pop_back();
195  }
196  } else {
197  // key == __i->first
198  _data.pop_back();
199  }
200  return std::make_pair(__i, retval);
201  }
