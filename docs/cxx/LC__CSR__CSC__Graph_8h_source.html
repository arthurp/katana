149  void determineInEdgeIndices(EdgeIndData& dataBuffer) {
150  // counting outgoing edges in the tranpose graph by
151  // counting incoming edges in the original graph
152  katana::do_all(
153  katana::iterate(UINT64_C(0), BaseGraph::numEdges), [&](uint64_t e) {
154  auto dst = BaseGraph::edgeDst[e];
155  __sync_add_and_fetch(&(dataBuffer[dst]), 1);
156  });
157 
158  // prefix sum calculation of the edge index array
159  for (uint32_t n = 1; n < BaseGraph::numNodes; ++n) {
160  dataBuffer[n] += dataBuffer[n - 1];
161  }
162 
163  // copy over the new tranposed edge index data
164  inEdgeIndData.allocateInterleaved(BaseGraph::numNodes);
165  katana::do_all(
166  katana::iterate(UINT64_C(0), BaseGraph::numNodes),
167  [&](uint64_t n) { inEdgeIndData[n] = dataBuffer[n]; });
168  }
169 
176  void determineInEdgeDestAndData(EdgeIndData& dataBuffer) {
177  // after this block dataBuffer[i] will now hold number of edges that all
178  // nodes before the ith node have; used to determine where to start
179  // saving an edge for a node
180  if (BaseGraph::numNodes >= 1) {
181  dataBuffer[0] = 0;
182  katana::do_all(
183  katana::iterate(UINT64_C(1), BaseGraph::numNodes),
184  [&](uint64_t n) { dataBuffer[n] = inEdgeIndData[n - 1]; });
185  }
186 
187  // allocate edge dests and data
188  inEdgeDst.allocateInterleaved(BaseGraph::numEdges);
189 
190  if (!std::is_void<EdgeTy>::value) {
191  inEdgeData.allocateInterleaved(BaseGraph::numEdges);
192  }
193 
194  katana::do_all(
195  katana::iterate(UINT64_C(0), BaseGraph::numNodes), [&](uint64_t src) {
196  // e = start index into edge array for a particular node
197  uint64_t e = (src == 0) ? 0 : BaseGraph::edgeIndData[src - 1];
198 
199  // get all outgoing edges of a particular node in the non-transpose
200  // and convert to incoming
201  while (e < BaseGraph::edgeIndData[src]) {
202  // destination nodde
203  auto dst = BaseGraph::edgeDst[e];
204  // location to save edge
205  auto e_new = __sync_fetch_and_add(&(dataBuffer[dst]), 1);
206  // save src as destination
207  inEdgeDst[e_new] = src;
208  // edge data to "new" array
209  createEdgeData(e_new, e);
210  e++;
211  }
212  });
213  }
