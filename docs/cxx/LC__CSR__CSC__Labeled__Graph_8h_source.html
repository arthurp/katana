316  template <bool in_edges, enum FindEdgeWithLabelOptions opt>
317  std::optional<std::tuple<edge_iterator, edge_iterator, edge_iterator>>
318  FindFirstOrLastEdge(
319  const edge_iterator& begin, const edge_iterator& end,
320  GraphNode key) const {
321  edge_iterator l = begin;
322  edge_iterator r = end - 1;
323 
324  // If we're searching for the first edge, we stop at the beginning and need
325  // to check the previous edge. The opposite is true for the last edge.
326  bool finding_first = opt == FindEdgeWithLabelOptions::FindFirst;
327  edge_iterator limit = finding_first ? begin : (end - 1);
328  int direction = finding_first ? -1 : 1;
329 
330  // If we're finding the first element, we can narrow the search space for
331  // finding the last element based off of what we've seen.
332  edge_iterator last_begin_bound = begin;
333  edge_iterator last_end_bound = end;
334 
335  while (r >= l) {
336  edge_iterator mid = l + (r - l) / 2;
337  GraphNode value =
338  in_edges ? this->getInEdgeDst(mid) : this->getEdgeDst(mid);
339  if (value == key) {
340  // Update bounds for last only if we're searching for the first element.
341  if (finding_first) {
342  last_begin_bound = mid;
343  last_end_bound = r + 1;
344  }
345 
346  if (!(opt == FindEdgeWithLabelOptions::FindAny) && mid != limit) {
347  // check that mid - 1 is not key.
348  GraphNode adjacent_value = in_edges
349  ? this->getInEdgeDst(mid + direction)
350  : this->getEdgeDst(mid + direction);
351  if (adjacent_value != key) {
352  return std::make_tuple(mid, last_begin_bound, last_end_bound);
353  } else {
354  if (finding_first) {
355  r = mid - 1;
356  } else {
357  l = mid + 1;
358  }
359  }
360  } else {
361  return std::make_tuple(mid, last_begin_bound, last_end_bound);
362  }
363  }
364  if (value < key)
365  l = mid + 1;
366  else
367  r = mid - 1;
368  }
369  return std::nullopt;
370  }
