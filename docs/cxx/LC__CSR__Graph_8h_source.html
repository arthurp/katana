247  template <typename EdgeNumFnTy, typename EdgeDstFnTy, typename EdgeDataFnTy>
248  LC_CSR_Graph(
249  uint32_t _numNodes, uint64_t _numEdges, EdgeNumFnTy edgeNum,
250  EdgeDstFnTy _edgeDst, EdgeDataFnTy _edgeData)
251  : numNodes(_numNodes), numEdges(_numEdges) {
252  if (UseNumaAlloc) {
254  nodeData.allocateBlocked(numNodes);
255  edgeIndData.allocateBlocked(numNodes);
256  edgeDst.allocateBlocked(numEdges);
257  edgeData.allocateBlocked(numEdges);
259  this->outOfLineAllocateBlocked(numNodes, false);
260  } else {
261  nodeData.allocateInterleaved(numNodes);
262  edgeIndData.allocateInterleaved(numNodes);
263  edgeDst.allocateInterleaved(numEdges);
264  edgeData.allocateInterleaved(numEdges);
265  this->outOfLineAllocateInterleaved(numNodes);
266  }
267  for (size_t n = 0; n < numNodes; ++n) {
268  nodeData.constructAt(n);
269  }
270  uint64_t cur = 0;
271  for (size_t n = 0; n < numNodes; ++n) {
272  cur += edgeNum(n);
273  edgeIndData[n] = cur;
274  }
275  cur = 0;
276  for (size_t n = 0; n < numNodes; ++n) {
277  for (uint64_t e = 0, ee = edgeNum(n); e < ee; ++e) {
278  if (EdgeData::has_value)
279  edgeData.set(cur, _edgeData(n, e));
280  edgeDst[cur] = _edgeDst(n, e);
281  ++cur;
282  }
283  }
284  }

524  void transpose(const char* regionName = NULL) {
525  katana::StatTimer timer("TIMER_GRAPH_TRANSPOSE", regionName);
526  timer.start();
527 
528  EdgeDst edgeDst_old;
529  EdgeData edgeData_new;
530  EdgeIndData edgeIndData_old;
531  EdgeIndData edgeIndData_temp;
532 
533  if (UseNumaAlloc) {
534  edgeIndData_old.allocateBlocked(numNodes);
535  edgeIndData_temp.allocateBlocked(numNodes);
536  edgeDst_old.allocateBlocked(numEdges);
537  edgeData_new.allocateBlocked(numEdges);
538  } else {
539  edgeIndData_old.allocateInterleaved(numNodes);
540  edgeIndData_temp.allocateInterleaved(numNodes);
541  edgeDst_old.allocateInterleaved(numEdges);
542  edgeData_new.allocateInterleaved(numEdges);
543  }
544 
545  // Copy old node->index location + initialize the temp array
546  katana::do_all(
547  katana::iterate(UINT64_C(0), numNodes),
548  [&](uint64_t n) {
549  edgeIndData_old[n] = edgeIndData[n];
550  edgeIndData_temp[n] = 0;
551  },
552  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEINTDATA_COPY"));
553 
554  // get destination of edge, copy to array, and
555  katana::do_all(
556  katana::iterate(UINT64_C(0), numEdges),
557  [&](uint64_t e) {
558  auto dst = edgeDst[e];
559  edgeDst_old[e] = dst;
560  // counting outgoing edges in the tranpose graph by
561  // counting incoming edges in the original graph
562  __sync_add_and_fetch(&edgeIndData_temp[dst], 1);
563  },
564  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEINTDATA_INC"));
565 
566  // TODO is it worth doing parallel prefix sum?
567  // prefix sum calculation of the edge index array
568  for (uint32_t n = 1; n < numNodes; ++n) {
569  edgeIndData_temp[n] += edgeIndData_temp[n - 1];
570  }
571 
572  // copy over the new tranposed edge index data
573  katana::do_all(
574  katana::iterate(UINT64_C(0), numNodes),
575  [&](uint64_t n) { edgeIndData[n] = edgeIndData_temp[n]; },
576  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEINTDATA_SET"));
577 
578  // edgeIndData_temp[i] will now hold number of edges that all nodes
579  // before the ith node have
580  if (numNodes >= 1) {
581  edgeIndData_temp[0] = 0;
582  katana::do_all(
583  katana::iterate(UINT64_C(1), numNodes),
584  [&](uint64_t n) { edgeIndData_temp[n] = edgeIndData[n - 1]; },
585  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEINTDATA_TEMP"));
586  }
587 
588  katana::do_all(
589  katana::iterate(UINT64_C(0), numNodes),
590  [&](uint64_t src) {
591  // e = start index into edge array for a particular node
592  uint64_t e = (src == 0) ? 0 : edgeIndData_old[src - 1];
593 
594  // get all outgoing edges of a particular node in the
595  // non-transpose and convert to incoming
596  while (e < edgeIndData_old[src]) {
597  // destination nodde
598  auto dst = edgeDst_old[e];
599  // location to save edge
600  auto e_new = __sync_fetch_and_add(&(edgeIndData_temp[dst]), 1);
601  // save src as destination
602  edgeDst[e_new] = src;
603  // copy edge data to "new" array
604  edgeDataCopy(edgeData_new, edgeData, e_new, e);
605  e++;
606  }
607  },
608  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEDST"));
609 
610  // if edge weights, then overwrite edgeData with new edge data
611  if (EdgeData::has_value) {
612  katana::do_all(
613  katana::iterate(UINT64_C(0), numEdges),
614  [&](uint64_t e) { edgeDataCopy(edgeData, edgeData_new, e, e); },
615  katana::no_stats(), katana::loopname("TRANSPOSE_EDGEDATA_SET"));
616  }
617 
618  timer.stop();
619  }

635  template <
636  typename E = EdgeTy,
637  std::enable_if_t<!std::is_same<E, void>::value, int>* = nullptr>
638  void constructFrom(
639  FileGraph& graph, unsigned tid, unsigned total,
640  const bool readUnweighted = false) {
641  // at this point memory should already be allocated
642  auto r =
643  graph
644  .divideByNode(
645  NodeData::size_of::value + EdgeIndData::size_of::value +
646  LC_CSR_Graph::size_of_out_of_line::value,
647  EdgeDst::size_of::value + EdgeData::size_of::value, tid, total)
648  .first;
649 
650  this->setLocalRange(*r.first, *r.second);
651 
652  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei; ++ii) {
653  nodeData.constructAt(*ii);
654  edgeIndData[*ii] = *graph.edge_end(*ii);
655 
656  this->outOfLineConstructAt(*ii);
657 
658  for (FileGraph::edge_iterator nn = graph.edge_begin(*ii),
659  en = graph.edge_end(*ii);
660  nn != en; ++nn) {
661  if (readUnweighted) {
662  edgeData.set(*nn, {});
663  } else {
664  constructEdgeValue(graph, nn);
665  }
666  edgeDst[*nn] = graph.getEdgeDst(nn);
667  }
668  }
669  }
