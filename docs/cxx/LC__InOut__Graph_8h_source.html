171  this->acquireNode(N, mflag);
172  if (!asymmetric) {

174  for (edge_iterator ii = this->raw_begin(N), ei = this->raw_end(N);

176  this->acquireNode(this->getEdgeDst(ii), mflag);
177  }
178  }
179  return in_edge_iterator(this->raw_begin(N));
180  } else {
181  if (katana::shouldLock(mflag)) {
182  for (typename InGraph::edge_iterator

184  ei = inGraph.raw_end(inGraphNode(N));
185  ii != ei; ++ii) {
186  this->acquireNode(
187  nodeFromId(inGraph.getId(inGraph.getEdgeDst(ii))), mflag);

256  void sortInEdgesByDst(GraphNode N, MethodFlag mflag = MethodFlag::WRITE) {
257  this->acquireNode(N, mflag);
258  if (!asymmetric) {
259  typedef EdgeSortValue<GraphNode, edge_data_type> EdgeSortVal;
260  std::sort(
261  this->edge_sort_begin(N), this->edge_sort_end(N),
262  [=](const EdgeSortVal& e1, const EdgeSortVal& e2) {
263  return e1.dst < e2.dst;
264  });
265  } else {
266  typedef EdgeSortValue<
267  typename InGraph::GraphNode, typename InGraph::edge_data_type>
268  InEdgeSortVal;
269  std::sort(
270  inGraph.edge_sort_begin(inGraphNode(N)),
271  inGraph.edge_sort_end(inGraphNode(N)),
272  [=](const InEdgeSortVal& e1, const InEdgeSortVal& e2) {
273  return e1.dst < e2.dst;
274  });
