333  void allocateFrom(FileGraph& graph) {
334  numNodes = graph.size();
335  numEdges = graph.sizeEdges();
336 
337  if (UseNumaAlloc) {
338  nodeData.allocateBlocked(numNodes);
339  edgeData.allocateBlocked(numEdges);

341  } else {

343  edgeData.allocateInterleaved(numEdges);
344  this->outOfLineAllocateInterleaved(numNodes);
345  }

348  void constructFrom(FileGraph& graph, unsigned tid, unsigned total) {
349  auto r = graph
350  .divideByNode(
351  NodeData::size_of::value +
352  LC_InlineEdge_Graph::size_of_out_of_line::value,
353  EdgeData::size_of::value, tid, total)
354  .first;
355 
356  EdgeInfo* curEdge = edgeData.data() + *graph.edge_begin(*r.first);
357 
358  this->setLocalRange(*r.first, *r.second);
359 
360  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei; ++ii) {
361  nodeData.constructAt(*ii);
362  this->outOfLineConstructAt(*ii);
363  nodeData[*ii].edgeBegin() = curEdge;
364  for (FileGraph::edge_iterator nn = graph.edge_begin(*ii),
365  en = graph.edge_end(*ii);
366  nn != en; ++nn) {
367  constructEdgeValue(graph, nn, curEdge);
368  setEdgeDst(nodeData, curEdge, graph.getEdgeDst(nn));
369  ++curEdge;
370  }
371  nodeData[*ii].edgeEnd() = curEdge;
372  }
373  }
