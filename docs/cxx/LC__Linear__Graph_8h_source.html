320  void allocateFrom(FileGraph& graph, const ReadGraphAuxData&) {
321  numNodes = graph.size();
322  numEdges = graph.sizeEdges();
323  if (UseNumaAlloc) {
324  data.allocateLocal(
325  sizeof(NodeInfo) * numNodes * 2 + sizeof(EdgeInfo) * numEdges);
326  nodes.allocateLocal(numNodes);
327  this->outOfLineAllocateLocal(numNodes);
328  } else {
329  data.allocateInterleaved(
330  sizeof(NodeInfo) * numNodes * 2 + sizeof(EdgeInfo) * numEdges);
331  nodes.allocateInterleaved(numNodes);
332  this->outOfLineAllocateInterleaved(numNodes);
333  }

336  void constructNodesFrom(
337  FileGraph& graph, unsigned tid, unsigned total, const ReadGraphAuxData&) {
338  auto r = graph
339  .divideByNode(
340  Nodes::size_of::value + 2 * sizeof(NodeInfo) +
341  LC_Linear_Graph::size_of_out_of_line::value,
342  sizeof(EdgeInfo), tid, total)
343  .first;
344 
345  this->setLocalRange(*r.first, *r.second);
346  NodeInfo* curNode = reinterpret_cast<NodeInfo*>(data.data());
347 
348  size_t id = *r.first;
349  size_t edges = *graph.edge_begin(*r.first);
350  size_t bytes = edges * sizeof(EdgeInfo) + 2 * (id + 1) * sizeof(NodeInfo);
351  curNode += bytes / sizeof(NodeInfo);
352  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei;
353  ++ii, ++id) {
354  nodes.constructAt(*ii);
355  new (curNode) NodeInfo();
356  // curNode->construct();
357  curNode->setId(id);
358  curNode->numEdges =
359  std::distance(graph.edge_begin(*ii), graph.edge_end(*ii));
360  nodes[*ii] = curNode;
361  curNode = curNode->next();
362  }
363  }
364 
365  void constructEdgesFrom(
366  FileGraph& graph, unsigned tid, unsigned total, const ReadGraphAuxData&) {
367  auto r = graph
368  .divideByNode(
369  Nodes::size_of::value + 2 * sizeof(NodeInfo) +
370  LC_Linear_Graph::size_of_out_of_line::value,
371  sizeof(EdgeInfo), tid, total)
372  .first;
373 
374  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei; ++ii) {
375  EdgeInfo* edge = nodes[*ii]->edgeBegin();
376  for (FileGraph::edge_iterator nn = graph.edge_begin(*ii),
377  en = graph.edge_end(*ii);
378  nn != en; ++nn) {
379  constructEdgeValue(graph, nn, edge);
380  edge->dst = nodes[graph.getEdgeDst(nn)];
381  ++edge;
382  }
383  }
384  }
