436  template <typename... Args>
437  GraphNode createNode(int nedges, Args&&... args) {
438  NodeInfo* N = &nodes.emplace(std::forward<Args>(args)...);
439  acquireNode(N, MethodFlag::WRITE);
440  EdgeHolder*& local_edges = *edgesL.getLocal();
441 
442  // Allocate space for a new EdgeHolder object if necessary
443  if (!local_edges ||
444  std::distance(local_edges->begin, local_edges->end) < nedges) {
445  EdgeHolder* old = local_edges;
446  // FIXME: this seems to leak
447  size_t size = allocSize();
448  void* block = pagePoolAlloc();
449  local_edges = reinterpret_cast<EdgeHolder*>(block);
450  local_edges->next = old;
451 
452  size -= sizeof(EdgeHolder);
453  block = reinterpret_cast<char*>(block) + sizeof(EdgeHolder);
454 
455  if (!std::align(
456  std::alignment_of_v<EdgeInfo>, sizeof(EdgeInfo), block, size)) {
457  KATANA_DIE("not enough space for EdgeInfo");
458  }
459 
460  local_edges->begin = reinterpret_cast<EdgeInfo*>(block);
461  local_edges->end = local_edges->begin;
462  local_edges->end += size / sizeof(EdgeInfo);
463  if (std::distance(local_edges->begin, local_edges->end) < nedges) {
464  KATANA_DIE("not enough space for EdgeInfo");
465  }
466  }
467 
468  // Set the memory aside for the new node in the edge holder object
469  N->edgeBegin = N->edgeEnd = local_edges->begin;
470  local_edges->begin += nedges;
471 #ifndef NDEBUG
472  N->trueEdgeEnd = local_edges->begin;
473 #endif
474  return GraphNode(N);
475  }

485  template <typename... Args>
486  edge_iterator addEdge(
487  GraphNode src, GraphNode dst, katana::MethodFlag mflag, Args&&... args) {
488  // katana::checkWrite(mflag, true);
489  acquireNode(src, mflag);

491  if (it == src->edgeEnd) {
492  it->dst = dst;

494  src->edgeEnd++;

496  }

579  void constructNodesFrom(
580  FileGraph& graph, unsigned tid, unsigned total, ReadGraphAuxData& aux) {
581  // get the portion of the graph that this thread is responsible for
582  // creating
583  auto r =
584  graph
585  .divideByNode(
586  sizeof(NodeInfo) + LC_Morph_Graph::size_of_out_of_line::value,
587  sizeof(EdgeInfo), tid, total)
588  .first;
589 
590  // create nodes of portion we are responsible for only
591  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei; ++ii) {
592  aux[*ii] =
593  createNode(std::distance(graph.edge_begin(*ii), graph.edge_end(*ii)));
594  }
595  }
596 
608  void constructEdgesFrom(
609  FileGraph& graph, unsigned tid, unsigned total,
610  const ReadGraphAuxData& aux) {
611  auto r =
612  graph
613  .divideByNode(
614  sizeof(NodeInfo) + LC_Morph_Graph::size_of_out_of_line::value,
615  sizeof(EdgeInfo), tid, total)
616  .first;
617 
618  for (FileGraph::iterator ii = r.first, ei = r.second; ii != ei; ++ii) {
619  for (FileGraph::edge_iterator nn = graph.edge_begin(*ii),
620  en = graph.edge_end(*ii);
621  nn != en; ++nn) {
622  constructEdgeValue(graph, nn, aux[*ii], aux[graph.getEdgeDst(nn)]);
623  }
624  }
625  }
