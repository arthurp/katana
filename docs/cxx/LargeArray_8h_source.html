38 template <typename T>
39 class LargeArray {
40  enum class AllocType { Blocked, Local, Interleaved, Floating };
41 
42  LAptr real_data_;
43  T* data_{};
44  size_t size_{};
45 
46  void Allocate(size_t n, AllocType t) {
47  KATANA_LOG_DEBUG_ASSERT(!data_);
48  size_ = n;
49  switch (t) {
50  case AllocType::Blocked:
51  real_data_ = largeMallocBlocked(n * sizeof(T), activeThreads);
52  break;
53  case AllocType::Interleaved:
54  real_data_ = largeMallocInterleaved(n * sizeof(T), activeThreads);
55  break;
56  case AllocType::Local:
57  real_data_ = largeMallocLocal(n * sizeof(T));
58  break;
59  case AllocType::Floating:
60  real_data_ = largeMallocFloating(n * sizeof(T));
61  break;
62  default:
63  KATANA_LOG_DEBUG_ASSERT(false);
64  };
65 
66  data_ = reinterpret_cast<T*>(real_data_.get());
67  }

162  template <typename RangeArray>
163  void allocateSpecified(size_type num, RangeArray& ranges) {
164  KATANA_LOG_DEBUG_ASSERT(!data_);
165 
166  real_data_ =
167  largeMallocSpecified(num * sizeof(T), activeThreads, ranges, sizeof(T));
168 
169  size_ = num;
170  data_ = reinterpret_cast<T*>(real_data_.get());
171  }

174  template <typename... Args>
175  void construct(Args&&... args) {
176  for (T *ii = data_, *ei = data_ + size_; ii != ei; ++ii) {
177  new (ii) T(std::forward<Args>(args)...);
178  }
179  }
180 
181  template <typename... Args>
182  void constructAt(size_type n, Args&&... args) {
183  new (&data_[n]) T(std::forward<Args>(args)...);
184  }
185 
187  template <typename... Args>
188  void create(size_type n, Args&&... args) {
189  allocateInterleaved(n);

191  }
192 
193  void deallocate() {
194  real_data_.reset();
