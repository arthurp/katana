389  void erase(iterator ii) {
390  if (SortedNeighbors) {
391  // For sorted case remove the element, moving following
392  // elements back to fill the space.
393  edges.erase(ii);
394  } else {
395  // We don't need to preserve the order, so move the last edge
396  // into this place and then remove last edge.
397  *ii = edges.back();
398  edges.pop_back();
399  }

405  void erase(gNode* N, bool inEdge = false) {
406  iterator ii = find(N, inEdge);
407  if (ii != end())
408  edges.erase(ii);
409  }
410 
414  iterator find(gNode* N, bool inEdge = false) {
415  iterator ii, ei = edges.end();
416  // find starting point to start search
417  if (SortedNeighbors) {
418  KATANA_LOG_DEBUG_ASSERT(std::is_sorted(
419  edges.begin(), edges.end(),
420  [=](const EdgeInfo& e1, const EdgeInfo& e2) {
421  return e1.first() < e2.first();
422  }));
423  ii =
424  std::lower_bound(edges.begin(), edges.end(), N, first_lt<gNode*>());
425  } else {
426  ii = edges.begin();
427  }
428 
429  first_eq_and_valid<gNode*> checker(N);
430  ii = std::find_if(ii, ei, checker);
431  while (ii != ei && ii->isInEdge() != inEdge) {
432  ++ii;
433  ii = std::find_if(ii, ei, checker);
434  };
435  return ii;
436  }

448  template <typename... Args>
449  iterator createEdge(gNode* N, EdgeTy* v, bool inEdge, Args&&... args) {
450  iterator ii;
451  if (SortedNeighbors) {
452  // If neighbors are sorted, find appropriate insertion point.
453  // Insert before first neighbor that is too far.
454  ii =
455  std::upper_bound(edges.begin(), edges.end(), N, first_lt<gNode*>());
456  } else {
457  ii = edges.end();
458  }
459 
460  return edges.insert(
461  ii, EdgeInfo(N, v, inEdge, std::forward<Args>(args)...));
462  }

468  template <typename... Args>
469  iterator createEdgeWithReuse(
470  gNode* N, EdgeTy* v, bool inEdge, Args&&... args) {
471  // First check for holes
472  iterator ii;
473  iterator ei;
474  if (SortedNeighbors) {
475  // If neighbors are sorted, find acceptable range for insertion.
476  ii =
477  std::lower_bound(edges.begin(), edges.end(), N, first_lt<gNode*>());
478  ei = std::upper_bound(ii, edges.end(), N, first_lt<gNode*>());
479  } else {
480  // If not sorted, we can insert anywhere in the list.
481  ii = edges.begin();
482  ei = edges.end();
483  }
484  ii = std::find_if(ii, ei, first_not_valid());
485  if (ii != ei) {
486  // FIXME: We could move elements around (short distances).
487  *ii = EdgeInfo(N, v, inEdge, std::forward<Args>(args)...);
488  return ii;
489  }
490  return edges.insert(
491  ei, EdgeInfo(N, v, inEdge, std::forward<Args>(args)...));
492  }

828  GraphNode src, GraphNode dst,
829  katana::MethodFlag mflag = MethodFlag::WRITE) {
830  KATANA_LOG_DEBUG_ASSERT(src);
831  KATANA_LOG_DEBUG_ASSERT(dst);
832  src->acquire(mflag);
833  typename gNodeTypes::iterator ii = src->find(dst), ei = src->end();
