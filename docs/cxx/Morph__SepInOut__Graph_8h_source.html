351  void erase(iterator ii, bool inEdge = false) {
352  auto& edgelist = (inEdge) ? in_edges : edges;
353  if (SortedNeighbors) {
354  // For sorted case remove the element, moving following
355  // elements back to fill the space.
356  edgelist.erase(ii);
357  } else {
358  // We don't need to preserve the order, so move the last edge
359  // into this place and then remove last edge.
360  *ii = edgelist.back();
361  edgelist.pop_back();
362  }
363  }
364 
365  void erase(gNode* N, bool inEdge = false) {
366  iterator ii = find(N, inEdge);
367  erase(ii, inEdge);
368  }
369 
370  iterator find(gNode* N, bool inEdge = false) {
371  auto& edgelist = (inEdge) ? in_edges : edges;
372  iterator ii, ei = edgelist.end();
373  if (SortedNeighbors) {
374  KATANA_LOG_DEBUG_ASSERT(std::is_sorted(
375  edgelist.begin(), edgelist.end(),
376  [=](const EdgeInfo& e1, const EdgeInfo& e2) {
377  return e1.first() < e2.first();
378  }));
379  ii = std::lower_bound(
380  edgelist.begin(), edgelist.end(), N, first_lt<gNode*>());
381  } else {
382  ii = edgelist.begin();
383  }
384 
385  first_eq_and_valid<gNode*> checker(N);
386  ii = std::find_if(ii, ei, checker);
387  while (ii != ei && ii->isInEdge() != inEdge) {
388  ++ii;
389  ii = std::find_if(ii, ei, checker);
390  };
391  return ii;
392  }

399  template <typename... Args>
400  iterator createEdge(gNode* N, EdgeTy* v, bool inEdge, Args&&... args) {
401  iterator ii;
402  auto& edgelist = (inEdge) ? in_edges : edges;
403  if (SortedNeighbors) {
404  // If neighbors are sorted, find appropriate insertion point.
405  // Insert before first neighbor that is too far.
406  ii = std::upper_bound(
407  edgelist.begin(), edgelist.end(), N, first_lt<gNode*>());
408  } else
409  ii = edgelist.end();
410  return edgelist.insert(
411  ii, EdgeInfo(N, v, inEdge, std::forward<Args>(args)...));
412  }

414  template <typename... Args>
415  iterator createEdgeWithReuse(
416  gNode* N, EdgeTy* v, bool inEdge, Args&&... args) {
417  auto& edgelist = (inEdge) ? in_edges : edges;
418  // Morph check for holes
419  iterator ii, ei;
420  if (SortedNeighbors) {
421  // If neighbors are sorted, find acceptable range for insertion.
422  ii = std::lower_bound(
423  edgelist.begin(), edgelist.end(), N, first_lt<gNode*>());
424  ei = std::upper_bound(ii, edgelist.end(), N, first_lt<gNode*>());
425  } else {
426  // If not sorted, we can insert anywhere in the list.
427  ii = edgelist.begin();
428  ei = edgelist.end();
429  }
430  ii = std::find_if(ii, ei, first_not_valid());
431  if (ii != ei) {
432  // FIXME: We could move elements around (short distances).
433  *ii = EdgeInfo(N, v, inEdge, std::forward<Args>(args)...);
434  return ii;
435  }
436  return edgelist.insert(
437  ei, EdgeInfo(N, v, inEdge, std::forward<Args>(args)...));
438  }
