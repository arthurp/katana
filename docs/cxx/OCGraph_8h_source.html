361  segment_type computeSegment(size_t startNode, size_t numEdges) {
362  typedef typename OCFileGraph::edge_offset_iterator edge_offset_iterator;
363 
364  segment_type ret;
365 
366  edge_offset_iterator outStart = outGraph.edge_offset_begin();
367  edge_offset_iterator outEnd = outGraph.edge_offset_end();
368  std::advance(outStart, startNode);
369  if (outStart == outEnd) {
370  ret.nodeBegin = ret.nodeEnd = iterator(0);
371  return ret;
372  }
373  edge_offset_iterator outNext =
374  std::lower_bound(outStart + 1, outEnd, *outStart + numEdges);
375  ptrdiff_t outNodes = std::distance(outStart, outNext);
376 
377  edge_offset_iterator inStart = inGraph->edge_offset_begin();
378  edge_offset_iterator inEnd = inGraph->edge_offset_end();
379  std::advance(inStart, startNode);
380  edge_offset_iterator inNext =
381  std::lower_bound(inStart + 1, inEnd, *inStart + numEdges);
382  ptrdiff_t inNodes = std::distance(inStart, inNext);
383 
384  ptrdiff_t nodes = std::min(outNodes, inNodes);
385 
386  ret.nodeBegin = iterator(startNode);
387  ret.nodeEnd = iterator(startNode + nodes);
388  return ret;
389  }
