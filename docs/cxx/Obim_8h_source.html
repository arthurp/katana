46 template <typename T, typename Index>
47 class OrderedByIntegerMetricData<T, Index, true> {
48 protected:
49  struct ThreadData {
50  bool hasWork;
51  std::deque<std::pair<Index, T>> stored;

56  OrderedByIntegerMetricData() : barrier(GetBarrier(activeThreads)) {}

58  bool hasStored(ThreadData& p, Index idx) {
59  for (auto& e : p.stored) {
60  if (e.first == idx) {
61  std::swap(e, p.stored.front());
62  return true;
63  }
64  }
65  return false;
66  }
67 
68  std::optional<T> popStored(ThreadData& p, Index idx) {
69  std::optional<T> item;
70  for (auto ii = p.stored.begin(), ei = p.stored.end(); ii != ei; ++ii) {
71  if (ii->first == idx) {
72  item = ii->second;
73  p.stored.erase(ii);
74  break;
75  }
76  }
77  return item;
78  }
79 };

262  bool updateLocal(ThreadData& p) {
263  if (p.lastMasterVersion != masterVersion.load(std::memory_order_relaxed)) {
264  for (;
265  p.lastMasterVersion < masterVersion.load(std::memory_order_relaxed);
266  ++p.lastMasterVersion) {

268  p.local[logEntry.first] = logEntry.second;
269  KATANA_LOG_DEBUG_ASSERT(logEntry.second);

276  KATANA_ATTRIBUTE_NOINLINE
277  std::optional<T> slowPop(ThreadData& p) {
278  bool localLeader = ThreadPool::isLeader();
279  Index msS = this->earliest;
280 
281  updateLocal(p);
282 
283  if (BSP && !UseMonotonic) {
284  msS = p.scanStart;
285  if (localLeader) {
286  for (unsigned i = 0; i < activeThreads; ++i) {
287  Index o = data.getRemote(i)->scanStart;
288  if (this->compare(o, msS))
289  msS = o;
290  }
291  } else {
292  Index o = data.getRemote(ThreadPool::getLeader())->scanStart;
293  if (this->compare(o, msS))
294  msS = o;
295  }
296  }
297 
298  for (auto ii = p.local.lower_bound(msS), ei = p.local.end(); ii != ei;
299  ++ii) {
300  std::optional<T> item;
301  if ((item = ii->second->pop())) {
302  p.current = ii->second;
303  p.curIndex = ii->first;
304  p.scanStart = ii->first;
305  return item;
306  }
307  }
308 
309  return std::nullopt;
310  }

312  KATANA_ATTRIBUTE_NOINLINE
313  CTy* slowUpdateLocalOrCreate(ThreadData& p, Index i) {
314  // update local until we find it or we get the write lock
315  do {
316  updateLocal(p);
317  auto it = p.local.find(i);
318  if (it != p.local.end())
319  return it->second;
320  } while (!masterLock.try_lock());
321  // we have the write lock, update again then create
322  updateLocal(p);
323  auto it = p.local.find(i);
324  CTy* C2 = (it != p.local.end()) ? it->second : nullptr;
325  if (!C2) {
326  C2 = new CTy();
327  p.local[i] = C2;
328  p.lastMasterVersion = masterVersion.load(std::memory_order_relaxed) + 1;
329  masterLog.push_back(std::make_pair(i, C2));
330  masterVersion.fetch_add(1);
331  }
332  masterLock.unlock();
333  return C2;
334  }
335 
336  inline CTy* updateLocalOrCreate(ThreadData& p, Index i) {
337  // Try local then try update then find again or else create and update the
338  // master log
339  auto it = p.local.find(i);
340  if (it != p.local.end())
341  return it->second;
342  // slowpath
343  return slowUpdateLocalOrCreate(p, i);
344  }

358  void push(const value_type& val) {
359  Index index = indexer(val);
360  ThreadData& p = *data.getLocal();
361 
362  KATANA_LOG_DEBUG_ASSERT(!UseMonotonic || this->compare(p.curIndex, index));
363 
364  // Fast path
365  if (index == p.curIndex && p.current) {
366  p.current->push(val);
367  return;
368  }
369 
370  // Slow path
371  CTy* C = updateLocalOrCreate(p, index);
372  if (BSP && this->compare(index, p.scanStart))
373  p.scanStart = index;
374  // Opportunistically move to higher priority work
375  if (!UseBarrier && this->compare(index, p.curIndex)) {
376  p.curIndex = index;
377  p.current = C;
378  }
379  C->push(val);
380  }

393  std::optional<value_type> pop() {
394  // Find a successful pop
395  ThreadData& p = *data.getLocal();
396  CTy* C = p.current;
397 
398  if (this->hasStored(p, p.curIndex))
399  return this->popStored(p, p.curIndex);
400 
401  if (!UseBarrier && BlockPeriod &&
402  ((p.numPops++ & ((1 << BlockPeriod) - 1)) == 0))
403  return slowPop(p);
404 
405  std::optional<value_type> item;
406  if (C && (item = C->pop()))
407  return item;
408 
409  if (UseBarrier)
410  return item;
411 
412  // Slow path
413  return slowPop(p);
414  }

416  template <bool Barrier = UseBarrier>
417  auto empty() -> typename std::enable_if<Barrier, bool>::type {
418  std::optional<value_type> item;
419  ThreadData& p = *data.getLocal();
420 
421  // try to pop from global worklist
422  item = slowPop(p);
423  if (item)
424  p.stored.push_back(std::make_pair(p.curIndex, *item));
425 
426  // check if there are thread-local work items
427  if (!p.stored.empty()) {
428  Index storedIndex = this->identity;
429  for (auto& e : p.stored) {
430  if (this->compare(e.first, storedIndex)) {
431  storedIndex = e.first;
432  }
433  }
434  p.curIndex = storedIndex;
435  p.current = p.local[storedIndex];
436  }
437  p.hasWork = !p.stored.empty();
438 
439  this->barrier.Wait();
440 
441  // align with the earliest level from threads that have works
442  bool hasWork = p.hasWork;
443  Index curIndex = (hasWork) ? p.curIndex : this->identity;
444  CTy* C = (hasWork) ? p.current : nullptr;
445 
446  for (unsigned i = 0; i < activeThreads; ++i) {
447  ThreadData& o = *data.getRemote(i);
448  if (o.hasWork && this->compare(o.curIndex, curIndex)) {
449  curIndex = o.curIndex;
450  C = o.current;
451  }
452  hasWork |= o.hasWork;
453  }
454 
455  this->barrier.Wait();
456 
457  p.current = C;
458  p.curIndex = curIndex;
459 
460  if (UseMonotonic) {
461  for (auto ii = p.local.begin(); ii != p.local.end();) {
462  bool toBreak = ii->second == C;
463  if (toBreak)
464  break;
465  ii = p.local.erase(ii);
466  }
467  }
468 
469  return !hasWork;
470  }
