41 // File format V1:
42 // version (1) {uint64_t LE}
43 // EdgeType size {uint64_t LE}
44 // numNodes {uint64_t LE}
45 // numEdges {uint64_t LE}
46 // outindexs[numNodes] {uint64_t LE} (outindex[nodeid] is index of first edge
47 // for nodeid + 1 (end interator. node 0 has an implicit start iterator of 0.
48 // outedges[numEdges] {uint32_t LE}
49 // potential padding (32bit max) to Re-Align to 64bits
50 // EdgeType[numEdges] {EdgeType size}

76  uint64_t outIndexs(uint64_t node) {
77  std::lock_guard<decltype(lock)> lg(lock);
78  std::streamoff pos = (4 + node) * sizeof(uint64_t);
79 
80  // move to correct position in file
81  if (locEdgeDst != pos) {
82  numSeeksEdgeDst++;
83  fileEdgeDst.seekg(pos, fileEdgeDst.beg);
84  locEdgeDst = pos;
85  }
86 
87  // read the value
88  uint64_t retval;
89  try {
90  fileEdgeDst.read(reinterpret_cast<char*>(&retval), sizeof(uint64_t));
91  } catch (const std::ifstream::failure& e) {
92  std::cerr << "Exception while reading edge destinations:" << e.what()
93  << "\n";
94  std::cerr << "IO error flags: EOF " << fileEdgeDst.eof() << " FAIL "
95  << fileEdgeDst.fail() << " BAD " << fileEdgeDst.bad() << "\n";
96  }
97 
98  // metadata update
99  auto numBytesRead = fileEdgeDst.gcount();
100  KATANA_LOG_DEBUG_ASSERT(numBytesRead == sizeof(uint64_t));
101  locEdgeDst += numBytesRead;
102  numBytesReadEdgeDst += numBytesRead;
103 
104  return retval;
105  }

107  uint64_t outEdges(uint64_t edge) {
108  std::lock_guard<decltype(lock)> lg(lock);
109  std::streamoff pos = (4 + numNodes) * sizeof(uint64_t) +
110  edge * (v2 ? sizeof(uint64_t) : sizeof(uint32_t));
111 
112  // move to correct position
113  if (locIndex != pos) {
114  numSeeksIndex++;
115  fileIndex.seekg(pos, fileEdgeDst.beg);
116  locIndex = pos;
117  }
118 
119  // v2 reads 64 bits, v1 reads 32 bits
120  if (v2) {
121  uint64_t retval;
122  try {
123  fileIndex.read(reinterpret_cast<char*>(&retval), sizeof(uint64_t));
124  } catch (const std::ifstream::failure& e) {
125  std::cerr << "Exception while reading index:" << e.what() << "\n";
126  std::cerr << "IO error flags: EOF " << fileIndex.eof() << " FAIL "
127  << fileIndex.fail() << " BAD " << fileIndex.bad() << "\n";
128  }
129 
130  auto numBytesRead = fileIndex.gcount();
131  KATANA_LOG_DEBUG_ASSERT(numBytesRead == sizeof(uint64_t));
132  locIndex += numBytesRead;
133  numBytesReadIndex += numBytesRead;
134  return retval;
135  } else {
136  uint32_t retval;
137  try {
138  fileIndex.read(reinterpret_cast<char*>(&retval), sizeof(uint32_t));
139  } catch (const std::ifstream::failure& e) {
140  std::cerr << "Exception while reading index:" << e.what() << "\n";
141  std::cerr << "IO error flags: EOF " << fileIndex.eof() << " FAIL "
142  << fileIndex.fail() << " BAD " << fileIndex.bad() << "\n";
143  }
144 
145  auto numBytesRead = fileIndex.gcount();
146  KATANA_LOG_DEBUG_ASSERT(numBytesRead == sizeof(uint32_t));
147  locIndex += numBytesRead;
148  numBytesReadIndex += numBytesRead;
149  return retval;
150  }

153  template <typename T>
154  T edgeData(uint64_t edge) {
155  KATANA_LOG_DEBUG_ASSERT(sizeof(T) <= sizeEdgeData);
156  std::lock_guard<decltype(lock)> lg(lock);
157  std::streamoff pos = (4 + numNodes) * sizeof(uint64_t) +
158  numEdges * (v2 ? sizeof(uint64_t) : sizeof(uint32_t));
159 
160  // align + move to correct position
161  pos = (pos + 7) & ~7;
162  pos += edge * sizeEdgeData;
163 
164  if (locEdgeData != pos) {
165  numSeeksEdgeData++;
166  fileEdgeData.seekg(pos, fileEdgeDst.beg);
167  locEdgeData = pos;
168  }
169 
170  T retval;
171  try {
172  fileEdgeData.read(reinterpret_cast<char*>(&retval), sizeof(T));
173  } catch (const std::ifstream::failure& e) {
174  std::cerr << "Exception while reading edge data:" << e.what() << "\n";
175  std::cerr << "IO error flags: EOF " << fileEdgeData.eof() << " FAIL "
176  << fileEdgeData.fail() << " BAD " << fileEdgeData.bad() << "\n";
177  }
178 
179  auto numBytesRead = fileEdgeData.gcount();
180  KATANA_LOG_DEBUG_ASSERT(numBytesRead == sizeof(T));
181  locEdgeData += numBytesRead;
182  numBytesReadEdgeData += numBytesRead;
183  /*fprintf(stderr, "READ:: %ld[", edge);
184  for(int i=0; i<sizeof(T); ++i){
185  fprintf(stderr, "%c", reinterpret_cast<char*>(&retval)[i]);
186  }
187  fprintf(stderr, "]");*/
188  return retval;
189  }
