136  void shrink_to_fit() {
137  if (size_ == 0) {
138  if (data_ != NULL) {
139  free(data_);
140  data_ = NULL;
141  capacity_ = 0;
142  }
143  } else if (size_ < capacity_) {
144  capacity_ = std::max(size_, kMinNonZeroCapacity);
145  _Tp* new_data_ = static_cast<_Tp*>(
146  realloc(reinterpret_cast<void*>(data_), capacity_ * sizeof(_Tp)));
147  KATANA_LOG_DEBUG_ASSERT(new_data_);
148  data_ = new_data_;
149  }
150  }
151 
152  void reserve(size_t n) {
153  if (n <= capacity_) {
154  return;
155  }
156 
157  // When reallocing, don't pay for elements greater than size_
158  shrink_to_fit();
159 
160  // reset capacity_ because its previous value need not be a power-of-2
161  capacity_ = kMinNonZeroCapacity;
162  // increase capacity in powers-of-2
163  while (capacity_ < n) {
164  capacity_ <<= 1;
165  }
166 
167  _Tp* new_data_ = static_cast<_Tp*>(
168  realloc(reinterpret_cast<void*>(data_), capacity_ * sizeof(_Tp)));
169  KATANA_LOG_DEBUG_ASSERT(new_data_);
170  data_ = new_data_;
171  }
172
