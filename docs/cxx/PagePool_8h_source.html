70  void* allocFromOS() {
71  void* ptr = katana::allocPages(1, true);
72  KATANA_LOG_DEBUG_ASSERT(ptr);
73  auto tid = katana::ThreadPool::getTID();
74  counts[tid] += 1;
75  std::lock_guard<katana::SimpleLock> lg(mapLock);
76  ownerMap[ptr] = tid;
77  return ptr;
78  }

103  void* pageAlloc() {
104  auto tid = katana::ThreadPool::getTID();
105  HeadPtr& hp = pool[tid].data;
106  if (hp.getValue()) {
107  hp.lock();
108  FreeNode* h = hp.getValue();
109  if (h) {
110  hp.unlock_and_set(h->next);
111  freeCounts[tid] -= 1;
112  return h;
113  }
114  hp.unlock();
115  }
116  return allocFromOS();
117  }
118 
119  void pageFree(void* ptr) {
120  KATANA_LOG_DEBUG_ASSERT(ptr);
121  mapLock.lock();
122  KATANA_LOG_DEBUG_ASSERT(ownerMap.count(ptr));
123  int i = ownerMap[ptr];
124  freeCounts[i] += 1;
125  mapLock.unlock();
126  HeadPtr& hp = pool[i].data;
127  hp.lock();
128  FreeNode* nh = reinterpret_cast<FreeNode*>(ptr);
129  nh->next = hp.getValue();
130  hp.unlock_and_set(nh);
131  }
132 
133  void pagePreAlloc() { pageFree(allocFromOS()); }
134 };
