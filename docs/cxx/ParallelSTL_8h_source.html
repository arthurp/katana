70 template <class InputIterator, class Predicate>
71 InputIterator
72 find_if(InputIterator first, InputIterator last, Predicate pred) {
73  typedef find_if_helper<InputIterator, Predicate> HelperTy;
74  typedef typename HelperTy::AccumulatorTy AccumulatorTy;
75  typedef katana::PerSocketChunkFIFO<256> WL;
76  AccumulatorTy accum;
77  HelperTy helper(accum, pred);
78  for_each(
79  katana::iterate(

81  helper, katana::disable_conflict_detection(), katana::no_pushes(),
82  katana::parallel_break(), katana::wl<WL>());

115  template <class RandomAccessIterator, class Context>
116  void operator()(
117  std::pair<RandomAccessIterator, RandomAccessIterator> bounds,
118  Context& ctx) {
119  if (std::distance(bounds.first, bounds.second) <= 1024) {
120  std::sort(bounds.first, bounds.second, comp);
121  } else {
122  typedef
123  typename std::iterator_traits<RandomAccessIterator>::value_type VT;
124  RandomAccessIterator pivot = choose_rand(bounds.first, bounds.second);
125  VT pv = *pivot;
126  pivot = std::partition(
127  bounds.first, bounds.second,
128  std::bind(comp, std::placeholders::_1, pv));
129  // push the lower bit
130  if (bounds.first != pivot)
131  ctx.push(std::make_pair(bounds.first, pivot));
132  // adjust the upper bit
133  pivot = std::find_if(
134  pivot, bounds.second,
135  std::bind(neq_to<VT>(comp), std::placeholders::_1, pv));
136  // push the upper bit
137  if (bounds.second != pivot)
138  ctx.push(std::make_pair(pivot, bounds.second));
139  }
140  }
141 };

143 template <typename RandomAccessIterator, class Predicate>
144 std::pair<RandomAccessIterator, RandomAccessIterator>
145 dual_partition(
146  RandomAccessIterator first1, RandomAccessIterator last1,
147  RandomAccessIterator first2, RandomAccessIterator last2, Predicate pred) {
148  typedef std::reverse_iterator<RandomAccessIterator> RI;
149  RI first3(last2), last3(first2);
150  while (true) {
151  while (first1 != last1 && pred(*first1))
152  ++first1;
153  if (first1 == last1)
154  break;
155  while (first3 != last3 && !pred(*first3))
156  ++first3;
157  if (first3 == last3)
158  break;
159  std::swap(*first1++, *first3++);
160  }
161  return std::make_pair(first1, first3.base());
162 }

164 template <typename RandomAccessIterator, class Predicate>
165 struct partition_helper {
166  typedef std::pair<RandomAccessIterator, RandomAccessIterator> RP;
167  struct partition_helper_state {
168  RandomAccessIterator first, last;
169  RandomAccessIterator rfirst, rlast;
170  SimpleLock Lock;
171  Predicate pred;
172  typename std::iterator_traits<RandomAccessIterator>::difference_type
173  BlockSize() {
174  return 1024;
175  }
176 
177  partition_helper_state(
178  RandomAccessIterator f, RandomAccessIterator l, Predicate p)
179  : first(f), last(l), rfirst(l), rlast(f), pred(p) {}
180  RP takeHigh() {
181  Lock.lock();
182  unsigned BS = std::min(BlockSize(), std::distance(first, last));
183  last -= BS;
184  RandomAccessIterator rv = last;
185  Lock.unlock();
186  return std::make_pair(rv, rv + BS);
187  }
188  RP takeLow() {
189  Lock.lock();
190  unsigned BS = std::min(BlockSize(), std::distance(first, last));
191  RandomAccessIterator rv = first;
192  first += BS;
193  Lock.unlock();
194  return std::make_pair(rv, rv + BS);
195  }
196  void update(RP low, RP high) {
197  Lock.lock();
198  if (low.first != low.second) {
199  rfirst = std::min(rfirst, low.first);
200  rlast = std::max(rlast, low.second);
201  }
202  if (high.first != high.second) {
203  rfirst = std::min(rfirst, high.first);
204  rlast = std::max(rlast, high.second);
205  }
206  Lock.unlock();
207  }
208  };
209 
210  partition_helper(partition_helper_state* s) : state(s) {}
211 
212  partition_helper_state* state;
213 
214  void operator()(unsigned, unsigned) {
215  RP high, low;
216  do {
217  RP parts = dual_partition(
218  low.first, low.second, high.first, high.second, state->pred);
219  low.first = parts.first;
220  high.second = parts.second;
221  if (low.first == low.second)
222  low = state->takeLow();
223  if (high.first == high.second)
224  high = state->takeHigh();
225  } while (low.first != low.second && high.first != high.second);
226  state->update(low, high);
227  }
228 };

230 template <class RandomAccessIterator, class Predicate>
231 RandomAccessIterator
232 partition(
233  RandomAccessIterator first, RandomAccessIterator last, Predicate pred) {
234  if (std::distance(first, last) <= 1024)
235  return std::partition(first, last, pred);
236  typedef partition_helper<RandomAccessIterator, Predicate> P;
237  typename P::partition_helper_state s(first, last, pred);
238  on_each(P(&s));
239  if (s.rfirst == first && s.rlast == last) { // perfect !
240  // abort();
241  return s.first;
242  }
243  return std::partition(s.rfirst, s.rlast, pred);
244 }

328 template <class InputIt, class OutputIt>
329 OutputIt
330 partial_sum(InputIt first, InputIt last, OutputIt d_first) {
331  using ValueType = typename std::iterator_traits<InputIt>::value_type;
332 
333  size_t sizeOfVector = std::distance(first, last);
334 
335  // only bother with parallel execution if vector is larger than some size
336  if (sizeOfVector >= 1024) {
337  const size_t numBlocks = katana::getActiveThreads();
338  const size_t blockSize = (sizeOfVector + numBlocks - 1) / numBlocks;
339  KATANA_LOG_DEBUG_ASSERT(numBlocks * blockSize >= sizeOfVector);
340 
341  std::vector<ValueType> localSums(numBlocks);
342 
343  // get the block sums
344  katana::do_all(
345  katana::iterate((size_t)0, numBlocks), [&](const size_t& block) {
346  // block start can extend past sizeOfVector if doesn't divide evenly
347  size_t blockStart = std::min(block * blockSize, sizeOfVector);
348  size_t blockEnd = std::min((block + 1) * blockSize, sizeOfVector);
349  KATANA_LOG_DEBUG_ASSERT(blockStart <= blockEnd);
350 
351  // partial accumulation of each block done now
352  std::partial_sum(
353  first + blockStart, first + blockEnd, d_first + blockStart);
354  // save the last number in this block: used for block prefix sum
355  if (blockEnd > 0) {
356  localSums[block] = *(d_first + blockEnd - 1);
357  } else {
358  localSums[block] = 0;
359  }
360  });
361 
362  // bulkPrefix[i] holds the starting sum of a particular block i
363  std::vector<ValueType> bulkPrefix(numBlocks);
364  // exclusive scan on local sums to get number to add to each block's
365  // set of indices
366  // Not using std::exclusive_scan because apparently it doesn't work for
367  // some compilers
368  ValueType runningSum = 0;
369  for (size_t i = 0; i < numBlocks; i++) {
370  bulkPrefix[i] = runningSum;
371  runningSum += localSums[i];
372  }
373 
374  katana::do_all(
375  katana::iterate((size_t)0, numBlocks), [&](const size_t& block) {
376  // add the sums of previous elements to blocks
377  ValueType numToAdd = bulkPrefix[block];
378  size_t blockStart = std::min(block * blockSize, sizeOfVector);
379  size_t blockEnd = std::min((block + 1) * blockSize, sizeOfVector);
380  KATANA_LOG_DEBUG_ASSERT(blockStart <= blockEnd);
381 
382  // transform applies addition to appropriate range
383  std::transform(
384  d_first + blockStart, d_first + blockEnd, d_first + blockStart,
385  [&](ValueType& val) { return val + numToAdd; });
386  });
387 
388  // return the iterator past the last element written
389  return d_first + sizeOfVector;
390  } else {
391  // vector is small; do it serially using standard library
392  return std::partial_sum(first, last, d_first);
393  }
394 }

396 template <class InputIt, class OutputIt, class UnaryOperation>
397 OutputIt
398 transform(
399  InputIt first, InputIt last, OutputIt d_first, UnaryOperation unary_op) {
400  using input_category =
401  typename std::iterator_traits<InputIt>::iterator_category;
402  using output_category =
403  typename std::iterator_traits<OutputIt>::iterator_category;
404  static_assert(
405  std::is_base_of_v<std::random_access_iterator_tag, input_category>,
406  "parallel transform is only supported for random access iterators");
407  static_assert(
408  std::is_base_of_v<std::random_access_iterator_tag, output_category>,
409  "parallel transform is only supported for random access iterators");
410 
411  using diff_type = typename std::iterator_traits<InputIt>::difference_type;
412 
413  on_each([&](unsigned tid, unsigned total) {
414  auto [begin, end] = block_range(first, last, tid, total);
415  diff_type offset = std::distance(first, begin);
416  std::transform(begin, end, d_first + offset, unary_op);
417  });
418 
419  return d_first + std::distance(first, last);
420 }

447 template <class InputIt, class OutputIt, class UnaryPredicate>
448 OutputIt
449 copy_if(InputIt first, InputIt last, OutputIt d_first, UnaryPredicate pred) {
450  using input_category =
451  typename std::iterator_traits<InputIt>::iterator_category;
452  using output_category =
453  typename std::iterator_traits<OutputIt>::iterator_category;
454  static_assert(
455  std::is_base_of_v<std::random_access_iterator_tag, input_category>,
456  "parallel copy_if is only supported for random access iterators");
457  static_assert(
458  std::is_base_of_v<std::random_access_iterator_tag, output_category>,
459  "parallel copy_if is only supported for random access iterators");
460 
461  using diff_type = typename std::iterator_traits<InputIt>::difference_type;
462 
463  // first on_each, set ranges
464  uint32_t num_threads = getActiveThreads();
465  std::vector<diff_type> prefix_sum(num_threads);
466  on_each([&](unsigned tid, unsigned total) {
467  auto [begin, end] = block_range(first, last, tid, total);
468  diff_type count = 0;
469  for (; first != last; ++first) {
470  if (pred(first)) {
471  count++;
472  }
473  }
474  prefix_sum[tid] = count;
475  });
476 
477  // calculate prefix sums
478  std::partial_sum(prefix_sum.begin(), prefix_sum.end(), prefix_sum.begin());
479 
480  // second on_each, do copy_if
481  on_each([&](unsigned tid, unsigned total) {
482  auto [begin, end] = block_range(first, last, tid, total);
483 
484  diff_type offset = tid == 0 ? 0 : prefix_sum[tid - 1];
485  OutputIt actual_end = std::copy_if(begin, end, d_first + offset, pred);
486 
487  KATANA_LOG_DEBUG_ASSERT(actual_end == prefix_sum[tid]);
488  });
489 
490  return d_first + prefix_sum.back();
491 }
