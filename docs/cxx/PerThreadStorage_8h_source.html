88 template <typename T>
89 class PerThreadStorage {
90  PerBackend* b;
91  unsigned offset;
92 
93  void destruct() {
94  if (offset == ~0U) {
95  return;
96  }
97 
98  for (unsigned n = 0; n < GetThreadPool().getMaxThreads(); ++n) {
99  reinterpret_cast<T*>(b->getRemote(n, offset))->~T();
100  }
101  b->deallocOffset(offset, sizeof(T));
102  offset = ~0U;
103  }
