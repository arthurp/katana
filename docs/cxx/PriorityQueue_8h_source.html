40  typename T, typename Cmp = std::less<T>,
41  typename Alloc = katana::FixedSizeAllocator<T>>
42 class ThreadSafeOrderedSet {
43  typedef std::set<T, Cmp, Alloc> Set;
44 
45 public:
46  typedef Set container_type;
47  typedef typename container_type::value_type value_type;
48  typedef typename container_type::reference reference;
49  typedef typename container_type::const_reference const_reference;

51  typedef typename container_type::size_type size_type;
52  typedef typename container_type::const_iterator iterator;
53  typedef typename container_type::const_iterator const_iterator;
54  typedef typename container_type::const_reverse_iterator reverse_iterator;
55  typedef
56  typename container_type::const_reverse_iterator const_reverse_iterator;
57  typedef katana::SimpleLock Lock_ty;
58 
59 private:
60  alignas(KATANA_CACHE_LINE_SIZE) Lock_ty mutex;
61  Set orderedSet;
62 
63 public:
64  template <
65  typename _T, typename _Cmp = std::less<_T>,
66  typename _Alloc = katana::FixedSizeAllocator<_T>>
67  using retype = ThreadSafeOrderedSet<
68  _T, _Cmp,
69  _Alloc>; // FIXME: loses Alloc and Cmp types
70 
71  explicit ThreadSafeOrderedSet(
72  const Cmp& cmp = Cmp(), const Alloc& alloc = Alloc())
73  : orderedSet(cmp, alloc) {}
74 
75  template <typename Iter>
76  ThreadSafeOrderedSet(
77  Iter b, Iter e, const Cmp& cmp = Cmp(), const Alloc& alloc = Alloc())
78  : orderedSet(cmp, alloc) {
79  for (; b != e; ++b) {
80  orderedSet.insert(*b);
81  }
82  }
83 
84  bool empty() const {
85  mutex.lock();
86  bool ret = orderedSet.empty();
87  mutex.unlock();
88 
89  return ret;
90  }
91 
92  size_type size() const {
93  mutex.lock();
94  size_type sz = orderedSet.size();
95  mutex.unlock();
96 
97  return sz;
98  }
99 
100  value_type top() const {
101  mutex.lock();
102  value_type x = *orderedSet.begin();
103  mutex.unlock();
104  return x;
105  }
106 
107  bool find(const value_type& x) const {
108  mutex.lock();
109  bool ret = (orderedSet.find(x) != orderedSet.end());
110  mutex.unlock();
111  return ret;
112  }
113 
114  // for compatibility with various stl types
115  inline void push_back(const value_type& x) { this->push(x); }
116  inline void insert(const value_type& x) { this->push(x); }
117 
118  bool push(const value_type& x) {
119  mutex.lock();
120  auto p = orderedSet.insert(x);
121  mutex.unlock();
122  return p.second;
123  }
124 
125  value_type pop() {
126  mutex.lock();
127  value_type x = *orderedSet.begin();
128  orderedSet.erase(orderedSet.begin());
129  mutex.unlock();
130  return x;
131  }
132 
133  bool remove(const value_type& x) {
134  mutex.lock();
135  bool ret = false;
136 
137  if (x == *orderedSet.begin()) {
138  orderedSet.erase(orderedSet.begin());
139  ret = true;
140  } else {
141  size_type s = orderedSet.erase(x);
142  ret = (s > 0);
143  }
144  mutex.unlock();
145 
146  return ret;
147  }
148 
149  void clear() {
150  mutex.lock();
151  orderedSet.clear();
152  mutex.unlock();
153  }
154 
155  const_iterator begin() const { return orderedSet.begin(); }
156  const_iterator end() const { return orderedSet.end(); }
157 };

182

184

191  RevCmp revCmp;
192 
193  const_reference top_internal() const {
194  KATANA_LOG_DEBUG_ASSERT(!container.empty());
195  return container.front();
196  }
197 
198  value_type pop_internal() {
199  KATANA_LOG_DEBUG_ASSERT(!container.empty());
200  std::pop_heap(container.begin(), container.end(), revCmp);
201 
202  value_type x = container.back();
203  container.pop_back();
204 
205  return x;
206  }

211 
212  template <typename Iter>

214  : container(b, e), revCmp(cmp) {
215  std::make_heap(container.begin(), container.end());
216  }
217 
218  bool empty() const { return container.empty(); }
219 
220  size_type size() const { return container.size(); }
221 
222  const_reference top() const { return container.front(); }
223 
224  // for compatibility with various stl types
225  inline void push_back(const value_type& x) { this->push(x); }
226  inline void insert(const value_type& x) { this->push(x); }
227 
228  void push(const value_type& x) {
229  container.push_back(x);
230  std::push_heap(container.begin(), container.end(), revCmp);
231  }
232 
233  value_type pop() {
234  KATANA_LOG_DEBUG_ASSERT(!container.empty());
235  std::pop_heap(container.begin(), container.end(), revCmp);
236 
237  value_type x = container.back();
238  container.pop_back();
239  return x;
240  }
241 
242  bool remove(const value_type& x) {
243  bool ret = false;
244 
245  // TODO: write a better remove method
246  if (x == top()) {
247  pop();
248  ret = true;
249  } else {
250  typename container_type::iterator nend =
251  std::remove(container.begin(), container.end(), x);
252 
253  ret = (nend != container.end());
254  container.erase(nend, container.end());
255 
256  std::make_heap(container.begin(), container.end(), revCmp);
257  }
258 
259  return ret;
260  }
261 
262  bool find(const value_type& x) const {
263  return (std::find(begin(), end(), x) != end());
264  }
265 
266  void clear() { container.clear(); }
267 
268  const_iterator begin() const { return container.begin(); }
269  const_iterator end() const { return container.end(); }
270 
271  void reserve(size_type s) { container.reserve(s); }
272 };
