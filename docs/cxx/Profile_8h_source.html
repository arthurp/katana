89  /* Initialize the PAPI library */

92  if (retval != PAPI_VER_CURRENT && retval > 0) {
93  KATANA_DIE(

95  }
96

100 
101  if ((retval = PAPI_thread_init(&papiGetTID)) != PAPI_OK) {
102  KATANA_DIE("PAPI thread init failed");

109  for (size_t i = 0; i < eventNames.size(); ++i) {
110  char buf[256];

120 template <typename V1, typename V2, typename V3>
121 void
122 papiStart(V1& eventSets, V2& papiResults, V3& papiEvents) {
123  katana::on_each([&](const unsigned tid, const unsigned numT) {
124  if (PAPI_register_thread() != PAPI_OK) {
125  KATANA_DIE("failed to register thread with PAPI");
126  }
127 
128  int& eventSet = *eventSets.getLocal();
129 
130  eventSet = PAPI_NULL;
131  papiResults.getLocal()->resize(papiEvents.size());
132 
133  if (PAPI_create_eventset(&eventSet) != PAPI_OK) {
134  KATANA_DIE("failed to init event set");
135  }
136  if (PAPI_add_events(eventSet, papiEvents.data(), papiEvents.size()) !=
137  PAPI_OK) {
138  KATANA_DIE("failed to add events");
139  }
140 
141  if (PAPI_start(eventSet) != PAPI_OK) {
142  KATANA_DIE("failed to start PAPI");
143  }
144  });
145 }

147 template <typename V1, typename V2, typename V3>
148 void
149 papiStop(V1& eventSets, V2& papiResults, V3& eventNames, const char* region) {
150  katana::on_each([&](const unsigned tid, const unsigned numT) {
151  int& eventSet = *eventSets.getLocal();
152 
153  if (PAPI_stop(eventSet, papiResults.getLocal()->data()) != PAPI_OK) {
154  KATANA_DIE("PAPI_stop failed");
155  }
156 
157  if (PAPI_cleanup_eventset(eventSet) != PAPI_OK) {
158  KATANA_DIE("PAPI_cleanup_eventset failed");
159  }
160 
161  if (PAPI_destroy_eventset(&eventSet) != PAPI_OK) {
162  KATANA_DIE("PAPI_destroy_eventset failed");
163  }
164 
165  assert(
166  eventNames.size() == papiResults.getLocal()->size() &&
167  "Both vectors should be of equal length");
168  for (size_t i = 0; i < eventNames.size(); ++i) {
169  katana::ReportStatSum(
170  region, eventNames[i], (*papiResults.getLocal())[i]);
171  }
172 
173  if (PAPI_unregister_thread() != PAPI_OK) {
174  KATANA_DIE("failed to un-register thread with PAPI");
175  }
176  });
177 }

191 template <typename F>
192 void
193 profilePapi(const F& func, const char* region) {
194  region = region ? region : "(NULL)";
195 
196  std::string eventNamesCSV;
197 
198  if (!katana::GetEnv("KATANA_PAPI_EVENTS", &eventNamesCSV) ||
199  eventNamesCSV.empty()) {
200  katana::gWarn(
201  "No Events specified. Set environment variable KATANA_PAPI_EVENTS");
202  katana::StatTimer timer(region);
203  katana::TimerGuard timer_guard(timer);
204  func();
205  return;
206  }
207 
208  internal::papiInit();
209 
210  std::vector<std::string> eventNames;
211 
212  internal::splitCSVstr(eventNamesCSV, eventNames);
213 
214  std::vector<int> papiEvents(eventNames.size());
215 
216  internal::decodePapiEvents(eventNames, papiEvents);
217 
218  katana::PerThreadStorage<int> eventSets;
219  katana::PerThreadStorage<std::vector<long_long>> papiResults;
220 
221  internal::papiStart(eventSets, papiResults, papiEvents);
222 
223  {
224  katana::StatTimer timer(region);
225  katana::TimerGuard timer_guard(timer);
226  func();
227  }
228 
229  internal::papiStop(eventSets, papiResults, eventNames, region);
230 }
