159 template <typename IterTy>
160 class SpecificRange {
161  // TODO(ddn): This range is not generalizable to all iterators; probably
162  // move into libcusp or provide a specific numeric range class to allow
163  // easy construction of local iteration spaces.
164 public:
165  typedef IterTy iterator;
166  typedef iterator local_iterator;
167  typedef typename std::iterator_traits<IterTy>::value_type value_type;
168 
169 private:
178  std::pair<local_iterator, local_iterator> local_pair() const {
179  uint32_t my_thread_id = ThreadPool::getTID();
180  uint32_t total_threads = activeThreads;
181 
182  iterator local_begin = thread_beginnings_[my_thread_id];
183  iterator local_end = thread_beginnings_[my_thread_id + 1];
184 
185  KATANA_LOG_DEBUG_ASSERT(local_begin <= local_end);
186 
187  if (thread_beginnings_[total_threads] == *global_end_ &&
188  *global_begin_ == 0) {
189  return std::make_pair(local_begin, local_end);
190  }
191 
192  // This path assumes that we were passed in thread_beginnings for the
193  // range 0 to last node, but the passed in range to execute is NOT the
194  // entire 0 to thread end range; therefore, work under the assumption that
195  // only some threads will execute things only if they "own" nodes in the
196  // range
197  iterator left = local_begin;
198  iterator right = local_end;
199 
200  // local = what this thread CAN do
201  // global = what this thread NEEDS to do
202 
203  // cutoff left and right if global begin/end require less than what we
204  // need
205  if (local_begin < global_begin_) {
206  left = global_begin_;
207  }
208  if (local_end > global_end_) {
209  right = global_end_;
210  }
211  // make sure range is sensible after changing left and right
212  if (left >= right || right <= left) {
213  left = right = global_end_;
214  }
215 
216  // Explanations/reasoning of possible cases
217  // [ ] = local ranges
218  // o = need to be included; global ranges = leftmost and rightmost circle
219  // x = not included
220  // ooooo[ooooooooxxxx]xxxxxx handled (left the same, right moved)
221  // xxxxx[xxxxxooooooo]oooooo handled (left moved, right the same)
222  // xxxxx[xxoooooooxxx]xxxxxx handled (both left/right moved)
223  // xxxxx[xxxxxxxxxxxx]oooooo handled (left will be >= right, set l = r)
224  // oooox[xxxxxxxxxxxx]xxxxxx handled (right will be <= left, set l = r)
225  // xxxxx[oooooooooooo]xxxxxx handled (left, right the same = local range)
226 
227  return std::make_pair(left, right);
228  }
229 
230  IterTy global_begin_;
231  IterTy global_end_;
232  std::vector<uint32_t> thread_beginnings_;
233 
234 public:
235  SpecificRange(IterTy begin, IterTy end, std::vector<uint32_t> thread_ranges)
236  : global_begin_(begin),
237  global_end_(end),
238  thread_beginnings_(std::move(thread_ranges)) {}
239 
240  iterator begin() const { return global_begin_; }
241  iterator end() const { return global_end_; }
242 
243  local_iterator local_begin() const { return local_pair().first; }
244  local_iterator local_end() const { return local_pair().second; }
245 };
