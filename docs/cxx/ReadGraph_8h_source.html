156 template <typename GraphTy>
157 void
158 readGraphDispatch(GraphTy& graph, read_with_aux_first_graph_tag, FileGraph& f) {
159  typedef typename GraphTy::ReadGraphAuxData Aux;
160  constexpr static const bool profile = false;
161 
162  katana::CondStatTimer<profile> TAlloc("AllocateAux");
163  TAlloc.start();
164  Aux* auxPtr = new Aux;
165  graph.allocateFrom(f, *auxPtr);
166  TAlloc.stop();
167 
168  katana::CondStatTimer<profile> TNode("ConstructNode");
169  TNode.start();
170  ReadGraphConstructNodesFrom<GraphTy, Aux> nodeReader(graph, f, *auxPtr);
171  katana::on_each(nodeReader);
172  TNode.stop();
173 
174  katana::CondStatTimer<profile> TOutEdge("ConstructOutEdge");
175  TOutEdge.start();
176  ReadGraphConstructOutEdgesFrom<GraphTy, Aux> outEdgeReader(graph, f, *auxPtr);
177  katana::on_each(outEdgeReader);
178  TOutEdge.stop();
179 
180  katana::CondStatTimer<profile> TInEdge("ConstructInEdge");
181  TInEdge.start();
182  ReadGraphConstructInEdgesFrom<GraphTy, Aux> inEdgeReader(graph, f, *auxPtr);
183  katana::on_each(inEdgeReader);
184  TInEdge.stop();
185 
186  katana::CondStatTimer<profile> TDestruct("DestructAux");
187  TDestruct.start();
188  delete auxPtr;
189  TDestruct.stop();
190 }
