64 template <typename T, typename MergeFunc, typename IdFunc>
65 class Reducible : public MergeFunc, public IdFunc {
66  katana::PerThreadStorage<T> data_;
67 
68  void merge(T& lhs, T&& rhs) {
69  T v{std::move(MergeFunc::operator()(lhs, std::move(rhs)))};
70  lhs = std::move(v);
71  }
72 
73  void merge(T& lhs, const T& rhs) { lhs = MergeFunc::operator()(lhs, rhs); }
74 
75 public:
76  using value_type = T;
77 
78  Reducible(MergeFunc merge_func, IdFunc id_func)
79  : MergeFunc(merge_func), IdFunc(id_func) {
80  for (unsigned i = 0; i < data_.size(); ++i) {
81  *(data_.getRemote(i)) = IdFunc::operator()();
82  }
83  }
84 
89  void update(T&& rhs) { merge(*data_.getLocal(), std::move(rhs)); }
90 
91  void update(const T& rhs) { merge(*data_.getLocal(), rhs); }
92 
96  T& getLocal() { return *data_.getLocal(); }
97 
101  T& reduce() {
102  T& lhs = *data_.getLocal();
103  for (unsigned int i = 1; i < data_.size(); ++i) {
104  T& rhs = *data_.getRemote(i);
105  merge(lhs, std::move(rhs));
106  rhs = IdFunc::operator()();
107  }
108 
109  return lhs;
110  }
111 
112  void reset() {
113  for (unsigned int i = 0; i < data_.size(); ++i) {
114  *data_.getRemote(i) = IdFunc::operator()();
115  }
116  }
117 };
