55  inline void lock() const {
56  int oldval = 0;
57  if (_lock.load(std::memory_order_relaxed))
58  goto slow_path;
59  if (!_lock.compare_exchange_weak(
60  oldval, 1, std::memory_order_acq_rel, std::memory_order_relaxed))

62  KATANA_LOG_DEBUG_ASSERT(is_locked());

66  }
