131 template <typename Stat_tp>
132 struct BasicStatMap {
133  using Stat = Stat_tp;
134  using Str = katana::gstl::Str;
135  using StrSet = katana::gstl::Set<Str>;
136  using StatMap = katana::gstl::Map<std::tuple<const Str*, const Str*>, Stat>;
137  using const_iterator = typename StatMap::const_iterator;
138 
139 protected:
140  StrSet symbols;
141  StatMap statMap;
142 
143  const Str* getOrInsertSymbol(const Str& s) {
144  auto p = symbols.insert(s);
145  return &*(p.first);
146  }
147 
148  const Str* getSymbol(const Str& s) const {
149  auto i = symbols.find(s);
150 
151  if (i == symbols.cend()) {
152  return nullptr;
153  } else {
154  return &(*i);
155  }
156  }
157 
158 public:
159  template <typename... Args>
160  Stat& getOrInsertStat(
161  const Str& region, const Str& category, Args&&... args) {
162  const Str* ln = getOrInsertSymbol(region);
163  const Str* cat = getOrInsertSymbol(category);
164 
165  auto tpl = std::make_tuple(ln, cat);
166 
167  auto p = statMap.emplace(tpl, Stat(std::forward<Args>(args)...));
168 
169  return p.first->second;
170  }
171 
172  const_iterator findStat(const Str& region, const Str& category) const {
173  const Str* ln = getSymbol(region);
174  const Str* cat = getSymbol(category);
175  auto tpl = std::make_tuple(ln, cat);
176 
177  auto i = statMap.find(tpl);
178 
179  return i;
180  }
181 
182  const Stat& getStat(const Str& region, const Str& category) const {
183  auto i = findStat(region, category);
184  KATANA_LOG_DEBUG_ASSERT(i != statMap.end());
185  return i->second;
186  }
187 
188  template <typename T, typename... Args>
189  void addToStat(
190  const Str& region, const Str& category, const T& val,
191  Args&&... statArgs) {
192  Stat& s =
193  getOrInsertStat(region, category, std::forward<Args>(statArgs)...);
194  s.add(val);
195  }
196 
197  const_iterator cbegin() const { return statMap.cbegin(); }
198  const_iterator cend() const { return statMap.cend(); }
199 
200  const Str& region(const const_iterator& i) const {
201  return *(std::get<0>(i->first));
202  }
203 
204  const Str& category(const const_iterator& i) const {
205  return *(std::get<1>(i->first));
206  }
207 
208  const Stat& stat(const const_iterator& i) const { return i->second; }
209 };

215 template <typename T>
216 struct VecStat : public VecStatMinMaxSum<T> {
217  StatTotal::Type m_totalTy;
218 
219  explicit VecStat(const StatTotal::Type& type) : m_totalTy(type) {}
220 
221  const StatTotal::Type& totalTy() const { return m_totalTy; }
222 
223  T total() const {
224  switch (m_totalTy) {
225  case StatTotal::SINGLE:
226  KATANA_LOG_DEBUG_ASSERT(this->values().size() > 0);
227  return this->values()[0];
228 
229  case StatTotal::TMIN:
230  return this->min();
231 
232  case StatTotal::TMAX:
233  return this->max();
234 
235  case StatTotal::TSUM:
236  return this->sum();
237 
238  case StatTotal::TAVG:
239  return this->avg();
240 
241  default:
242  KATANA_DIE("unreachable");
243  }
244  }
