97  void nextPoint(Point& p, int dim, int delta) {
98  KATANA_LOG_DEBUG_ASSERT(dim < numDims);
99  p[dim] += delta;
100  // account for overflow
101  while (p[dim] >= locks[dim].size()) {
102  p[dim] -= locks[dim].size();
103  }

137  Task* probeBlockWithLock(Point& start, int dim, size_t n) {
138  Point p = start;
139 
140  for (size_t i = 0; i < n; ++i) {
141  Task* t = getTask(p);
142 
143  KATANA_LOG_DEBUG_ASSERT(p[0] == t->coord[0]);
144  KATANA_LOG_DEBUG_ASSERT(p[1] == t->coord[1]);
145  KATANA_LOG_DEBUG_ASSERT(t->coord[0] < locks[0].size());
146  KATANA_LOG_DEBUG_ASSERT(t->coord[1] < locks[1].size());
147 
148  if (t->updates.relaxedLoad() < maxUpdates) {
149  if (std::try_lock(locks[0][t->coord[0]], locks[1][t->coord[1]]) < 0) {
150  if (t->updates.relaxedLoad() < maxUpdates) {
151  t->updates.relaxedAdd(1);
152  start = p;
153  return t;
154  }
155 
156  // TODO add to worklist
157  for (int i = 0; i < numDims; ++i) {
158  locks[i][t->coord[i]].unlock();
159  }
160  }
161  }
162 
163  nextPoint(p, dim, 1);
164  }
165 
166  failedProbes += 1;
167  return nullptr;
168  }

181  Task* probeBlockWithoutLock(Point& start, int dim, size_t n) {
182  Point p = start;
183 
184  for (size_t i = 0; i < n; ++i) {
185  Task* t = getTask(p);
186 
187  KATANA_LOG_DEBUG_ASSERT(p[0] == t->coord[0]);
188  KATANA_LOG_DEBUG_ASSERT(p[1] == t->coord[1]);
189  KATANA_LOG_DEBUG_ASSERT(t->coord[0] < locks[0].size());
190  KATANA_LOG_DEBUG_ASSERT(t->coord[1] < locks[1].size());
191 
192  if (t->updates.relaxedLoad() < maxUpdates) {
193  if (t->updates.value.fetch_add(1) < maxUpdates) {
194  // hasn't reached maxed updates at point of fetch
195  start = p;
196  return t;
197  }
198  }
199  nextPoint(p, dim, 1);
200  }
201 
202  failedProbes += 1;
203  return nullptr;
204  }

251  Task* nextBlock(Point& start, bool inclusive) {
252  Task* t;
253 
254  // repeats twice just to make sure there are actually no unused blocks
255  // TODO this method of termination detection is hacky and imprecise,
256  // find a better way
257  for (int times = 0; times < 2; ++times) {
258  Point limit{{locks[0].size(), locks[1].size()}};
259 
260  int inclusiveDelta = (inclusive && times == 0) ? 0 : 1;
261 
262  // First iteration (i.e. inclusive = true) is INCLUSIVE of start
263  // Otherwise, check the next blocks in the x and y direction for the
264  // next block
265  for (int i = 0; i < numDims; ++i) {
266  Point p = start;
267  nextPoint(p, i, inclusiveDelta);
268 
269  if ((t = probeBlock(p, i, limit[i] - inclusiveDelta))) {
270  start = p;
271  return t;
272  }
273  }
274 
275  // if the above for loop failed, it means all blocks in both directions
276  // (left->right, up->down) from current block from point are locked
277  // and/or all blocks have reached max updates
278  Point p = start;
279  // solution to above issue in comment = advance using diagonal and check
280  // from there
281  for (int i = 0; i < numDims; ++i) {
282  nextPoint(p, i, 1);
283  }
284 
285  // below will end up looping through entire grid looking for a block
286  // to work on; in some cases a block will be looped over more than once
287  // (see below TODO)
288  // TODO probably unoptimal: if any limit has hit 0, is it the case that
289  // the entire grid has been looked at already? This comment writer thinks
290  // the answer is yes in which case the below is doing extra work
291  while (std::any_of(
292  limit.begin(), limit.end(), [](size_t x) { return x > 0; })) {
293  for (int i = 0; i < numDims; ++i) {
294  if (limit[i] > 1 && (t = probeBlock(p, i, limit[i] - 1))) {
295  start = p;
296  return t;
297  }
298  }
299 
300  for (int i = 0; i < numDims; ++i) {
301  if (limit[i] > 0) {
302  limit[i] -= 1;
303  nextPoint(p, i, 1);
304  }
305  }
306  }
307  }
308 
309  return nullptr;
310  }

324  void executeBlock(
325  Function& fn, Task& task, typename std::enable_if<UseDense>::type* = 0) {
326  GetDst getDst{&g};
327 
328  for (auto ii = task.startX; ii != task.endX; ++ii) {
329  for (auto jj = g.begin() + task.startY,
330  ej = g.begin() + task.endYInclusive + 1;
331  jj != ej; ++jj) {
332  fn(*ii, *jj);
333  }
334  }

389  template <bool UseDense, typename Function>
390  void executeLoopOrig(Function fn, unsigned tid, unsigned total) {
391  Point numBlocks{{locks[0].size(), locks[1].size()}};
392  Point block;
393  Point start;
394 
395  // find out each thread's starting point; essentially what it is doing
396  // is assinging each thread to a block on the diagonal to begin with
397  for (int i = 0; i < numDims; ++i) {
398  block[i] = (numBlocks[i] + total - 1) / total; // blocks per thread
399  start[i] = std::min(block[i] * tid, numBlocks[i] - 1); // block to start
400  }
401 
402  unsigned coresPerSocket = katana::GetThreadPool().getMaxCores() /
403  katana::GetThreadPool().getMaxSockets();
404 
405  // if using locks, readjust start Y location of this thread to location of
406  // the thread's socket
407  if (useLocks) {
408  start = {
409  {start[0], std::min(
410  block[1] * katana::GetThreadPool().getSocket(tid) *
411  coresPerSocket,
412  numBlocks[1] - 1)}};
413  }
414 
415  Point p = start;
416 
417  for (int i = 0;; ++i) {
418  Task* t = nextBlock(p, i == 0);
419  // TODO: Replace with sparse worklist, etc.
420  if (!t)
421  break;
422 
423  executeBlock<UseDense>(fn, *t);
424 
425  // unlock the task block if using locks (next block returns the task with
426  // the block locked)
427  if (useLocks) {
428  for (int i = 0; i < numDims; ++i) {
429  locks[i][t->coord[i]].unlock();
430  }
431  }
432  }
433  }

462  void initializeTasks(
463  iterator firstX, iterator lastX, iterator firstY, iterator lastY,
464  size_t sizeX, size_t sizeY) {
465  const size_t numXBlocks =
466  (std::distance(firstX, lastX) + sizeX - 1) / sizeX;
467  const size_t numYBlocks =
468  (std::distance(firstY, lastY) + sizeY - 1) / sizeY;
469  const size_t numBlocks = numXBlocks * numYBlocks;
470 
471  // locks[0].create(numXBlocks);
472  // locks[1].create(numYBlocks);
473  // tasks.create(numBlocks);
474  locks[0].resize(numXBlocks);
475  locks[1].resize(numYBlocks);
476  tasks.resize(numBlocks);
477 
478  // TODO parallelize this?
479  // assign each block the X and Y that it is responsible for
480  for (size_t i = 0; i < numBlocks; ++i) {
481  Task& task = tasks[i];
482  task.coord = {{i % numXBlocks, i / numXBlocks}};
483  std::tie(task.startX, task.endX) =
484  katana::block_range(firstX, lastX, task.coord[0], numXBlocks);
485  iterator s;
486  iterator e;
487  std::tie(s, e) =
488  katana::block_range(firstY, lastY, task.coord[1], numYBlocks);
489  // Works for CSR graphs
490  task.startY = *s;
491  task.endYInclusive = *e - 1;
492  }
493  }
