61  template <typename... Args>
62  Block* alloc_block(Args&&... args) {
63  // Fixed size allocator can only allocate 1 object at a time of size
64  // sizeof(Block). Argument to allocate is always 1.
65  Block* b = heap.allocate(1);
66  return new (b) Block(std::forward<Args>(args)...);
67  }
68 
69  void free_block(Block* b) {
70  b->~Block();
71  heap.deallocate(b, 1);
72  }
74 
75  bool precondition() const {
76  return (num == 0 && first == NULL && last == NULL) ||
77  (num > 0 && first != NULL && last != NULL);
78  }
79 
80  Block* extend_first() {
81  Block* b = alloc_block();
82  b->next = first;
83  if (b->next)
84  b->next->prev = b;
85  first = b;
86  if (!last)
87  last = b;
88  return b;
89  }
90 
91  Block* extend_last() {
92  Block* b = alloc_block();
93  b->prev = last;
94  if (b->prev)
95  b->prev->next = b;
96  last = b;
97  if (!first)
98  first = b;
99  return b;
100  }
101 
102  void shrink(Block* b) {
103  if (b->next)
104  b->next->prev = b->prev;
105  if (b->prev)
106  b->prev->next = b->next;
107  if (b == first)
108  first = b->next;
109  if (b == last)
110  last = b->prev;
111  free_block(b);
112  }
113 
114  template <typename... Args>
115  std::pair<Block*, typename Block::iterator> emplace(
116  Block* b, typename Block::iterator ii, Args&&... args) {
117  ++num;
118  if (!b) {
119  // gdeque is empty or iteration == end
120  b = last;
121  if (!b || b->full())
122  b = extend_last();
123  ii = b->end();
124  } else if (b == first && ii == b->begin()) {
125  // iteration == begin
126  b = first;
127  if (!b || b->full())
128  b = extend_first();
129  ii = b->begin();
130  } else if (b->full()) {
131  auto d = std::distance(ii, b->end());
132  Block* n = alloc_block(
133  std::make_move_iterator(ii), std::make_move_iterator(b->end()));
134  for (; d > 0; --d)
135  b->pop_back();
136  ii = b->end();
137  n->next = b->next;
138  n->prev = b;
139  b->next = n;
140  if (b == last)
141  last = n;
142  }
143  unsigned boff = std::distance(b->begin(), ii);
144  b->emplace(ii, std::forward<Args>(args)...);
145  return std::make_pair(b, b->begin() + boff);
146  }
