94  template <typename HeapTy, bool C = Concurrent>
95  auto extend_first(HeapTy& heap) -> typename std::enable_if<C>::type {
96  Block* b = alloc_block(heap);
97  while (true) {
98  Block* f = first.load(std::memory_order_relaxed);
99  b->next = f;
100  if (first.compare_exchange_weak(f, b))
101  return;
102  }
103  }
