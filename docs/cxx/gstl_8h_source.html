252 template <
253  typename IterTy,
254  typename std::enable_if<!std::is_integral<IterTy>::value>::type* = nullptr>
255 std::pair<IterTy, IterTy>
256 block_range(IterTy b, IterTy e, unsigned id, unsigned num) {
257  size_t dist = std::distance(b, e);
258  size_t numper = std::max((dist + num - 1) / num, (size_t)1); // round up
259  size_t A = std::min(numper * id, dist);
260  size_t B = std::min(numper * (id + 1), dist);
261  std::advance(b, A);
262 
263  if (dist != B) {
264  e = b;
265  std::advance(e, B - A);
266  }
267 
268  return std::make_pair(b, e);
269 }

271 template <
272  typename IntTy,
273  typename std::enable_if<std::is_integral<IntTy>::value>::type* = nullptr>
274 std::pair<IntTy, IntTy>
275 block_range(IntTy b, IntTy e, unsigned id, unsigned num) {
276  IntTy dist = e - b;
277  IntTy numper = std::max((dist + num - 1) / num, (IntTy)1); // round up
278  IntTy A = std::min(numper * id, dist);
279  IntTy B = std::min(numper * (id + 1), dist);
280  b += A;
281  if (dist != B) {
282  e = b;
283  e += (B - A);
284  }
285  return std::make_pair(b, e);
286 }
